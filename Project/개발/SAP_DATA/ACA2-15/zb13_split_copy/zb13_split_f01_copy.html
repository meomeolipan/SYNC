FORM SELECT_DATA .
  SELECT A~CARRID
        A~CONNID
        A~FLDATE
        A~PRICE
        A~CURRENCY
        A~PLANETYPE
        A~SEATSMAX
        A~SEATSOCC
        A~PAYMENTSUM
        A~SEATSMAX_B
        A~SEATSOCC_B
        A~SEATSMAX_F
        A~SEATSOCC_F
        B~CARRNAME
        INTO CORRESPONDING FIELDS OF TABLE GT_DATA
        FROM SFLIGHT AS A
        INNER JOIN SCARR AS B
        ON A~CARRID = B~CARRID
        WHERE A~CARRID IN S_CAR
        AND A~CONNID IN S_CON
        AND A~FLDATE IN S_FLD.
ENDFORM.

FORM HANDLE_DOUBLE_CLICK USING P_E_ROW TYPE LVC_S_ROW
                                   P_E_COLUMN TYPE LVC_S_COL
                                   P_ES_ROW_NO TYPE LVC_S_ROID.
CLEAR GS_DATA.
READ TABLE GT_DATA INTO GS_DATA INDEX P_E_ROW-INDEX.

SELECT * FROM SBOOK INTO CORRESPONDING FIELDS OF TABLE GT_BOOK
  WHERE CARRID = GS_DATA-CARRID
  AND CONNID = GS_DATA-CONNID
  AND FLDATE = GS_DATA-FLDATE.

CLEAR GS_BOOK.
LOOP AT GT_BOOK INTO GS_BOOK.
  IF GS_BOOK-SMOKER = 'X'.
    "빨간색
    GS_CELL-FNAME = 'CUSTOMID'.

    APPEND GS_CELL TO GS_BOOK-CELL_COLOR.
    CLEAR GS_CELL.
   ENDIF.
*
   IF GS_BOOK-CANCELLED = 'X'.
     "주황색
     GS_BOOK-ROW_COLOR = 'C701'.
   ENDIF.


   modify GT_BOOK from GS_BOOK.
   CLEAR GS_BOOK.
  ENDLOOP.
