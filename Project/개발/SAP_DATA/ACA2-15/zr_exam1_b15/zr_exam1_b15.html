INITIALIZATION.

  SET TITLEBAR 'TITLE_1000'.

  LOOP AT SCREEN.
    IF screen-name = 'P_CNAME'.
      screen-input = 0.
    ENDIF.
    MODIFY SCREEN.
  ENDLOOP.

  MOVE TEXT-fb1 TO sscrfields-functxt_01.

  MOVE: 'I' TO s_scarr-sign,
        'BT' TO s_scarr-option,
        'AC' TO s_scarr-low,
        'AZ' TO s_scarr-high.

  APPEND s_scarr.
  CLEAR s_scarr.

  MOVE: 'I' TO s_scarr-sign,
        'EQ' TO s_scarr-option,
        'AA' TO s_scarr-low.

  APPEND s_scarr.
  CLEAR s_scarr.

  MOVE: 'E' TO s_scarr-sign,
        'EQ' TO s_scarr-option,
        'LT' TO s_scarr-low.

  APPEND s_scarr.

AT SELECTION-SCREEN.

  IF sscrfields-ucomm = 'FC01'.
    MESSAGE '기능키 구현 완료' TYPE 'I'.
  ENDIF.
