*& <--  p2        text
*&---------------------------------------------------------------------*
FORM make_data2 .
  LOOP AT gt_sbook INTO gs_sbook.

    IF gs_sbook-class = 'Y'.
      gs_sbook-tclass = 'Economy Class'.
    ELSEIF gs_sbook-class = 'C'.
      gs_sbook-tclass = 'Business Class'.
    ELSE.
      gs_sbook-tclass = 'First Class'.
    ENDIF.

    "row color
    IF gs_sbook-cancelled = 'X'.
      gs_sbook-color = 'C701'.
    ENDIF.

    "cell color
    IF gs_sbook-smoker = 'X'.
      gs_field_color-fname = 'CUSTOMID'.
      gs_field_color-color-col = 6.
      gs_field_color-color-int = 0.
      gs_field_color-color-inv = 0.
      APPEND gs_field_color TO gs_sbook-it_field_colors.
      CLEAR gs_field_color.
    ENDIF.

    MODIFY gt_sbook FROM gs_sbook.
    CLEAR gs_sbook.
  ENDLOOP.

ENDFORM.
