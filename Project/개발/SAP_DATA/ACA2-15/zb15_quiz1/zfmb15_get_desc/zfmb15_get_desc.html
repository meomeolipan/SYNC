FUNCTION zfmb15_get_desc.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" REFERENCE(IT_VBAK) TYPE ZGTY_VBAK
*" EXPORTING
*" REFERENCE(EV_DESC) TYPE ZDESC
*" EXCEPTIONS
*" NO_DATA
*"----------------------------------------------------------------------

* Global data declarations

  DATA: ls_vbak LIKE LINE OF it_vbak,
        test    TYPE vbak.

  LOOP AT it_vbak INTO ls_vbak.
    CASE ls_vbak-vbtyp.
      WHEN 'A'.
        ev_desc = 'Inquiry'.
      WHEN 'B'.
        ev_desc = 'Quotation'.
      WHEN OTHERS.
        ev_desc = 'Order'.
    ENDCASE.

  ENDLOOP.

  IF ev_desc is INITIAL.
    RAISE no_data.
  ENDIF.

ENDFUNCTION.
