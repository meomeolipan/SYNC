FUNCTION zfmb15_get_bezei.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" REFERENCE(IT_VBAK) TYPE ZGTY_VBAK
*" REFERENCE(IT_TVAKT) TYPE ZGTY_TVAKT
*" EXPORTING
*" REFERENCE(ET_BEZEI) TYPE TVAKT_BEZEI
*" EXCEPTIONS
*" NO_DATA
*"----------------------------------------------------------------------
* Global data declarations

  DATA:
    ls_vbak  LIKE LINE OF it_vbak,
    lt_vbak like it_vbak,
    lt_tvakt like it_tvakt,
    ls_tvakt like line of it_tvakt.


  LOOP AT it_vbak INTO ls_vbak.
     READ TABLE IT_TVAKT INTO ls_tvakt WITH KEY auart = ls_vbak-auart.
     et_bezei = ls_tvakt-bezei.
  ENDLOOP.

IF et_bezei is INITIAL.
  RAISE no_data.
ENDIF.


ENDFUNCTION.
