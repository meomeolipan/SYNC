MODULE set_icon OUTPUT.
  IF gs_member-mb_st EQ 'B'.
    icon_name = 'ICON_RED_LIGHT'.
    PERFORM icon_create USING icon_name CHANGING icon.
  ELSEIF gs_member-mb_st EQ 'A'.
    icon_name = 'ICON_GREEN_LIGHT'.
    PERFORM icon_create USING icon_name CHANGING icon.
  ELSE.
    CLEAR icon.
  ENDIF.

ENDMODULE.

MODULE get_push_status OUTPUT.
  IF gs_member-mb_st NE 'A' OR ztbook_b15-book_st NE 'A'.
    LOOP AT SCREEN.
      IF screen-name = 'PUSHBUTTON'.
        screen-input = 0.
      ENDIF.
      MODIFY SCREEN.
    ENDLOOP.
  ENDIF.
ENDMODULE.
