FUNCTION zb15_get_airline_info.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" REFERENCE(IV_CARRID) TYPE S_CARR_ID
*" EXPORTING
*" REFERENCE(ES_AIRLINE) TYPE ZSSCARR_B15
*" TABLES
*" ET_SPFLI TYPE ZTSPFLI_B15
*" EXCEPTIONS
*" NO_DATA
*"----------------------------------------------------------------------

* Global data declarations

  SELECT SINGLE * INTO es_airline
           FROM scarr
          WHERE carrid EQ iv_carrid.

  SELECT * INTO TABLE et_spfli
           FROM spfli
          WHERE carrid EQ iv_carrid.

  IF ( es_airline IS INITIAL AND et_spfli IS INITIAL ).
    RAISE no_data.
  ENDIF.

ENDFUNCTION.
