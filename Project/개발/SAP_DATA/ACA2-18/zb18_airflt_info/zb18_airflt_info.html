REPORT zb18_airflt_info.

PARAMETERS pa_c_id TYPE s_carr_id.

DATA gt_spfli TYPE ztspfli_b18.
DATA gs_spfli LIKE LINE OF gt_spfli.
DATA gs_scarr TYPE zsscarr_b18.

CALL FUNCTION 'ZB18_FLIGHT_GET_AIRLINE_INFO'
  EXPORTING
    iv_carrid  = pa_c_id
  IMPORTING
    es_airline = gs_scarr
  TABLES
    et_spfli   = gt_spfli
  EXCEPTIONS
    no_data    = 1
    OTHERS     = 2.

IF sy-subrc <> 0.
  IF sy-subrc = 1.
    WRITE '해당하는 데이터가 없습니다'.
  ENDIF.
ELSE.
  WRITE gs_scarr. " 항공사 정보 출력
  ULINE.
  PERFORM printschedule. " 항공 스케줄 출력
ENDIF.

FORM printschedule.
  LOOP AT gt_spfli INTO gs_spfli.
    WRITE: / gs_spfli-carrid,
             gs_spfli-connid,
             gs_spfli-Countryfr,
             gs_spfli-cityfrom,
             gs_spfli-airpfrom,
             gs_spfli-countryto,
             gs_spfli-cityto,
             gs_spfli-airpto,
             gs_spfli-fltime,
             gs_spfli-deptime,
             gs_spfli-arrtime.
  ENDLOOP.
ENDFORM.
