FORM get_data .
  SELECT a~mb_id a~mb_nm c~book_id c~book_nm b~rent_dt b~rplan_dt b~return_st b~return_dt
    INTO CORRESPONDING FIELDS OF TABLE gt_mem_rent
    FROM ztmember_b18 AS a
    INNER JOIN ztrent_b18 AS b
    ON a~mb_id = b~mb_id
    INNER JOIN ztbook_b18 AS c
    ON b~book_id = c~book_id
    WHERE a~mb_id IN s_memid
    AND b~rplan_dt IN s_return
    AND a~mb_st = p_memst.

  SORT gt_mem_rent BY mb_id book_id.

  SELECT * FROM ztmember_b18 INTO CORRESPONDING FIELDS OF TABLE gt_mem_return
    WHERE mb_id IN s_memid.

ENDFORM.
