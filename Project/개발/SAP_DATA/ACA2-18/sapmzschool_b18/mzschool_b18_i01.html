*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0110  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0110 INPUT.

  IF ok_code = 'APPLY'.
    select single * from ztclass_b18
      where studentno = ztstudent_b18-studentno and
            subjectno = ztsubject_b18-subjectno and
            applygrade = ztclass_b18-applygrade and
            applysemester = ztclass_b18-applysemester.

    if sy-subrc = 0.
      MESSAGE '해당 학기에 이미 신청이 되어있습니다.' TYPE 'W'.
    else.
* 신청이 안 되어 있는 경우
      ztclass_b18-studentno = ztstudent_b18-studentno.
      ztclass_b18-subjectno = ztsubject_b18-subjectno.
      ztclass_b18-applygrade = ztsubject_b18-opengrade.
      ztclass_b18-applysemester = ztsubject_b18-opensemester.
      ztclass_b18-last_changed_by = sy-uname.
      ztclass_b18-changed_on = sy-datum.
      ztclass_b18-changed_time = sy-uzeit.
      modify ztclass_b18.
    endif.
  elseif ok_code = 'BACK'.
    leave to screen 0.

  ELSEIF ok_code = 'FC1' OR ok_code = 'FC2'.
    my_tabstrip-activetab = ok_code.



  endif.
ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       text

MODULE user_command_0100 INPUT.
  select single * from ztstudent_b18
    where studentno = ztstudent_b18-studentno.

  select single * from ztsubject_b18
    where subjectno = ztsubject_b18-subjectno.
ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  EXIT  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE exit INPUT.
  CASE ok_code.
    WHEN 'CANCEL'.
      clear : ztstudent_b18,
              ztclass_b18,
              ztsubject_b18.
      LEAVE TO SCREEN 100.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
  endcase.
ENDMODULE.
