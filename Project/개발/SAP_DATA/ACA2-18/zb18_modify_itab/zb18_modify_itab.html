Description: Program ZB18_MODIFY_ITAB
*&---------------------------------------------------------------------*
*& Report ZB18_MODIFY_ITAB
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
REPORT zb18_modify_itab.

PARAMETERS : pa_carr TYPE s_carr_id.

DATA : BEGIN OF gs_sflight.
         INCLUDE STRUCTURE sflight.
DATA : tot_seatsmax TYPE s_seatsmax,
         tot_seatsocc TYPE s_seatsocc,
         icon type icon-id,
       END OF gs_sflight.

DATA : gt_sflight LIKE TABLE OF gs_sflight.

data : gv_line type i.

SELECT * INTO CORRESPONDING FIELDS OF TABLE gt_sflight
  FROM sflight WHERE carrid = pa_carr.

LOOP AT gt_sflight INTO gs_sflight.
  gs_sflight-tot_seatsmax = gs_sflight-seatsmax + gs_sflight-seatsmax_b + gs_sflight-seatsmax_f.
  gs_sflight-tot_seatsocc = gs_sflight-seatsocc + gs_sflight-seatsocc_b + gs_sflight-seatsocc_f.

  if gs_sflight-tot_seatsocc >= 300.
    gs_sflight-icon = icon_green_light.
  elseif gs_sflight-tot_seatsocc > 100.
    gs_sflight-icon = icon_yellow_light.
  else.
    gs_sflight-icon = icon_red_light.
  endif.

  MODIFY gt_sflight FROM gs_sflight.
ENDLOOP.

SORT gt_sflight BY carrid connid fldate.

gs_sflight-tot_seatsmax = 0.

LOOP AT gt_sflight INTO gs_sflight.
  WRITE: / gs_sflight-carrid, gs_sflight-connid, gs_sflight-fldate,
           gs_sflight-tot_seatsmax, gs_sflight-tot_seatsocc, gs_sflight-icon as icon.
ENDLOOP.

describe table gt_sflight lines gv_line.

write : / gv_line, '개 의 데이터가 검색되었습니다'.
