REPORT zb18_gugu.

PARAMETERS : pa_num TYPE i.
DATA : gv_result TYPE i.
DATA : gv_result_str TYPE string.

IF pa_num >= 2 AND pa_num <= 9.
  DO 9 TIMES.
    gv_result = pa_num * sy-index.
    gv_result_str = |{ pa_num } { '*' } { sy-index } { '=' } { gv_result }|.
*  WRITE :/ pa_num, '*', sy-index, '=', gv_result.
   WRITE :/ gv_result_str.
  ENDDO.
elseif pa_num is initial.
  write TEXT-002.
ELSE.
  WRITE TEXT-001.
ENDIF.
