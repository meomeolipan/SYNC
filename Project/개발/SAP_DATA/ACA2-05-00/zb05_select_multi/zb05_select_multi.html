REPORT ZB05_SELECT_MULTI.

DATA: gs_itab TYPE bc400_s_connection.

DATA: gt_itab LIKE TABLE OF gs_itab.

SELECT *
  INTO CORRESPONDING FIELDS OF TABLE gt_itab
  FROM spfli
  WHERE carrid = 'LH'.

  "공부해보니 order by carrid connid descending.을
  "select 문 뒤에 붙이면 아래 sort문을 쓸 필요가 없음.

SORT gt_itab BY carrid DESCENDING connid DESCENDING.

CLEAR gs_itab.

LOOP AT gt_itab INTO gs_itab.
  WRITE: / gs_itab-airpfrom,
         / gs_itab-airpto,
         / gs_itab-arrtime,
         / gs_itab-carrid,
         / gs_itab-cityfrom,
         / gs_itab-cityto,
         / gs_itab-connid,
         / gs_itab-deptime,
         / gs_itab-fltime.
  ULINE.
ENDLOOP.
