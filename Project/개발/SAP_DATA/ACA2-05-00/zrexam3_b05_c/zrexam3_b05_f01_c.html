FORM select_data .

  CASE 'X'.
    WHEN pa_r1.

      REFRESH gt_data.

      SELECT a~mb_id a~mb_nm
             b~book_id b~book_nm
             c~rent_dt c~rplan_dt c~return_st c~return_dt
        INTO CORRESPONDING FIELDS OF TABLE gt_data
        FROM ztrent_b05   AS c
        JOIN ztmember_b05 AS a ON c~mb_id = a~mb_id
        JOIN ztbook_b05   AS b ON c~book_id = b~book_id
        WHERE a~mb_id     IN so_id
          AND a~mb_st     EQ pa_st
          AND c~rplan_dt  IN so_rp
        ORDER BY a~mb_id b~book_id.

    WHEN pa_r2.

      REFRESH gt_split.

      SELECT a~mb_id a~mb_nm a~telno a~email
        INTO CORRESPONDING FIELDS OF TABLE gt_split
        FROM ztmember_b05  AS a
        JOIN ztrent_b05 AS b ON a~mb_id = b~mb_id
        WHERE a~mb_id     IN so_id
          AND a~mb_st     EQ pa_st
          AND b~rplan_dt  IN so_rp.

  ENDCASE.

ENDFORM.

*&---------------------------------------------------------------------*
FORM modify_data .

  REFRESH gt_display.

  IF gt_data IS NOT INITIAL.

    LOOP AT gt_data INTO gs_data.

      CLEAR gs_display.
      MOVE-CORRESPONDING gs_data TO gs_display.

      IF gs_display-return_st     EQ 'A'
                                  AND gs_display-return_dt > gs_display-rplan_dt
         OR gs_display-return_st  EQ 'A'
                                  AND sy-datum > gs_display-rplan_dt.

        gs_display-light = '1'. " 빨간색

      ELSEIF gs_display-return_st EQ 'A'
                                  AND gs_display-return_dt < gs_display-rplan_dt. " 조건 완벽하게 추가해야함

        gs_display-light = '2'. " 노란색

      ELSE.

        gs_display-light = '3'. " 초록색

      ENDIF.

      CALL FUNCTION 'STF4_GET_DOMAIN_VALUE_TEXT'
        EXPORTING
          iv_domname    = 'ZD_RST_B05'
          iv_value      = gs_display-return_st
        IMPORTING
          ev_value_text = gs_display-return_st_txt
        EXCEPTIONS
          OTHERS        = 1.
      IF sy-subrc <> 0 .
        MESSAGE i003 WITH '잘못된 접근입니다.'.
      ENDIF.

      APPEND gs_display TO gt_display.

      CLEAR gs_display.

    ENDLOOP.

  ENDIF.

  IF gt_return IS  NOT INITIAL.

    CLEAR gs_return.

    LOOP AT gt_return INTO gs_return.

      IF gs_return-rplan_dt < sy-datum.

        gs_return-fees = 300 * (  sy-datum - gs_return-rplan_dt ). " 클래스로도 구현 가능할 듯?

        gs_color-fname = 'FEES'.
        gs_color-color-col = col_negative.
        gs_color-color-int = 0.
        gs_color-color-inv = 1.

        APPEND gs_color TO gs_return-cell_color.

      ENDIF.

      gs_return-cuky = 'KRW'.

      MODIFY gt_return FROM gs_return TRANSPORTING fees cuky cell_color.

    ENDLOOP.

  ENDIF.

ENDFORM.
