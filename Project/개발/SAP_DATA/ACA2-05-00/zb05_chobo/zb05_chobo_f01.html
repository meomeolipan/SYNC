FORM select_data .

  CLEAR: gt_data, gt_display.

  SELECT a~carrid a~carrname
         b~connid b~countryfr b~countryto
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    FROM scarr AS a
    INNER JOIN spfli AS b ON  a~carrid EQ b~carrid   "join 만 써도 inner join 임.
    INNER JOIN sbook AS c ON  a~carrid EQ c~carrid
                          AND b~connid EQ c~connid
*    WHERE a~carrid EQ pa_car. " parameter가 아닌 select-option 이면 in 사용.
  WHERE a~carrid EQ pa_car
    AND b~connid IN so_con
    AND c~fldate IN so_fld.

ENDFORM.

FORM modify_data .

  CLEAR: gt_display.

  LOOP AT gt_data INTO gs_data.

    CLEAR: gs_display.

    MOVE-CORRESPONDING gs_data TO gs_display.

    APPEND gs_display TO gt_display.

  ENDLOOP.

ENDFORM.
