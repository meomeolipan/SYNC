*&---------------------------------------------------------------------*

INITIALIZATION.

AT SELECTION-SCREEN OUTPUT.
*  BREAK-POINT.
  IF p_sfl = 'X'.
    LOOP AT SCREEN.
      IF screen-group1 = 'SF'.
        screen-active = 1.
        IF screen-name = 'S_SF_CON'.
          screen-values_in_combo = '1'.
        ENDIF.
      ELSEIF screen-group1 = 'SP'.
        screen-active = 0.
      ENDIF.
      MODIFY SCREEN.
    ENDLOOP.
  ELSEIF p_spf = 'X'.
    LOOP AT SCREEN.
      IF screen-group1 = 'SP'.
        screen-active = 1.
        IF screen-name CS 'SP_CAR'.
          screen-intensified = '1'.
        ENDIF.
      ELSEIF screen-group1 = 'SF'.
        screen-active = 0.
      ENDIF.
      MODIFY SCREEN.
    ENDLOOP.
  ENDIF.
