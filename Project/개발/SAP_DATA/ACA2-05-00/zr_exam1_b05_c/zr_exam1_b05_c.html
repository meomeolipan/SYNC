REPORT zr_exam1_b05_c.

TABLES : sscrfields, saplane, sflight, scarr, sbuspart.

DATA: gs_sbuspart TYPE sbuspart,
      gt_flight   TYPE TABLE OF sflight.

CONSTANTS gc_mark TYPE c VALUE 'X'.

SELECTION-SCREEN FUNCTION KEY 1.

SELECTION-SCREEN BEGIN OF BLOCK b1 WITH FRAME TITLE TEXT-b01.
  PARAMETERS : p_car LIKE sflight-carrid MEMORY ID car DEFAULT 'AA'.
  SELECT-OPTIONS: so_car FOR sflight-carrid.
SELECTION-SCREEN END OF BLOCK b1.

SELECTION-SCREEN BEGIN OF BLOCK b2 WITH FRAME TITLE TEXT-b02 NO INTERVALS.
  SELECT-OPTIONS: so_ptype FOR saplane-planetype NO-EXTENSION.
  SELECT-OPTIONS: so_prod FOR saplane-producer.
  SELECT-OPTIONS: so_pcar FOR scarr-carrid.
SELECTION-SCREEN END OF BLOCK b2.


SELECTION-SCREEN BEGIN OF LINE.

  SELECTION-SCREEN COMMENT 1(10) TEXT-001 FOR FIELD p_carr.

  PARAMETERS : p_carr LIKE scarr-carrid.

  SELECTION-SCREEN COMMENT 17(15) TEXT-002 FOR FIELD p_cname.
  SELECT-OPTIONS: p_cname FOR scarr-carrname NO-EXTENSION NO INTERVALS.
  SELECTION-SCREEN POSITION POS_LOW.

SELECTION-SCREEN END OF LINE.

SELECTION-SCREEN SKIP.

SELECTION-SCREEN BEGIN OF BLOCK b3 WITH FRAME TITLE TEXT-b03.
  PARAMETERS : p_num LIKE sbuspart-buspartnum.

  SELECTION-SCREEN SKIP.

  SELECTION-SCREEN BEGIN OF BLOCK b4 WITH FRAME TITLE TEXT-b04.
    PARAMETERS: pa_cus RADIOBUTTON GROUP r1,
                pa_ag  RADIOBUTTON GROUP r1 DEFAULT 'X',
                pa_etc RADIOBUTTON GROUP r1.
  SELECTION-SCREEN END OF BLOCK b4.

SELECTION-SCREEN END OF BLOCK b3.

PARAMETERS: p_cb1 TYPE c AS CHECKBOX DEFAULT 'X',
            p_cb2 TYPE c AS CHECKBOX.

INITIALIZATION.

  MOVE: 'AA' TO so_car-low.
  APPEND so_car.
  CLEAR so_car.

  MOVE: 'AC' TO so_car-low,
        'AZ' TO so_car-high,
        'BT' TO so_car-option,
        'I' TO so_car-sign.
  APPEND so_car.
  CLEAR so_car.

  MOVE: 'LT' TO so_car-low,
        'EQ' TO so_car-option,
        'E' TO so_car-sign.
  APPEND so_car.
  CLEAR so_car.

  MOVE 'Call subscreen' TO sscrfields-functxt_01.

START-OF-SELECTION.

AT SELECTION-SCREEN OUTPUT.
  LOOP AT SCREEN.
    IF screen-name = 'P_CNAME-LOW' OR screen-name = 'P_CNAME-HIGH'.
      screen-input = 0.
    ENDIF.
    MODIFY SCREEN.
  ENDLOOP.

AT SELECTION-SCREEN ON p_num.
  LOOP AT SCREEN.
    SELECT SINGLE * FROM sbuspart INTO gs_sbuspart
          WHERE buspartnum = p_num.
    IF gs_sbuspart-buspatyp = 'FC'.
      pa_cus = 'X'.
    ELSEIF gs_sbuspart-buspatyp = 'TA'.
      pa_ag = 'X'.
    ELSE.
      pa_etc = 'X'.
    ENDIF.
  ENDLOOP.


  IF sscrfields-ucomm = 'FC01'.
    MESSAGE '기능키 구현 완료' TYPE 'I'.
  ENDIF.
