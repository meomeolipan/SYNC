CLASS lcl_screen_manager IMPLEMENTATION.

  METHOD handle_dynamic_screen.
    LOOP AT SCREEN.
      IF screen-group1 = 'BO'.
        IF iv_pa_r1 = 'X'.
          screen-active = 1. " 활성화
        ELSEIF iv_pa_r2 = 'X'.
          IF screen-name = 'SO_ID' OR screen-name = 'SO_ID-LOW' OR screen-name = 'SO_ID-HIGH'.
            screen-active = 1. " so_id는 활성화 유지
          ELSE.
            screen-active = 0. " 그 외 요소는 비활성화
          ENDIF.
        ENDIF.
      ENDIF.
      MODIFY SCREEN.
    ENDLOOP.
  ENDMETHOD.

ENDCLASS.
