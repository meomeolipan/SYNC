LOAD-OF-PROGRAM.

INITIALIZATION.

AT SELECTION-SCREEN OUTPUT.
  LOOP AT SCREEN.
    IF screen-group1 = 'Z1'.
      screen-active = gv_g1.
    ELSEIF  screen-group1 = 'Z2'.
      screen-active = gv_g2.
    ENDIF.
    MODIFY SCREEN.
  ENDLOOP.

AT SELECTION-SCREEN.
  CASE sy-ucomm.
    WHEN 'RAD'.
      IF p_zap = 'X'.
        gv_g1 = 1.
        gv_g2 = 0.
      ELSEIF p_zma = 'X'.
        gv_g1 = 0.
        gv_g2 = 1.
      ENDIF.
  ENDCASE.

START-OF-SELECTION.

  CASE gc_mark.
    WHEN p_zap.
      SELECT SINGLE * FROM zb05_comp INTO CORRESPONDING FIELDS OF gs_comp
        WHERE bukrs EQ p_buk
        AND zdate EQ p_zda
        AND zcode EQ p_zco
        AND zappro NE 'X'.
    WHEN p_zma.
      SELECT SINGLE * FROM zb05_comp INTO CORRESPONDING FIELDS OF gs_comp
        WHERE bukrs EQ p_buk
        AND zdate EQ p_zda
        AND zevetyp EQ p_zev
        AND zrelat EQ p_zre
        AND zappro NE 'X'.
    WHEN OTHERS.
      SELECT SINGLE * FROM zb05_comp INTO CORRESPONDING FIELDS OF gs_comp
        WHERE bukrs EQ p_buk
        AND zdate EQ p_zda
        AND zcode EQ p_zco
        AND zappro NE 'X'.
  ENDCASE.

  CASE gs_comp-zposition.
    WHEN 'A'.
      gs_comp-zposition_nm = '사원'.
    WHEN 'B'.
      gs_comp-zposition_nm = '대리'.
    WHEN 'C'.
      gs_comp-zposition_nm = '과장'.
    WHEN 'D'.
      gs_comp-zposition_nm = '부장'.
    WHEN 'Z'.
      gs_comp-zposition_nm = '인턴'.
  ENDCASE.

  CONCATENATE '경조금신청-' gs_comp-zname gs_comp-zposition_nm INTO gs_comp-zhtxt.
  gs_comp-zappro = 'X'.
  gs_comp-ernam = sy-uname. "생성자
  gs_comp-erdat = sy-datum. "생성일
  gs_comp-erzet = sy-timlo. "생성시간
  MODIFY zb05_comp FROM gs_comp.
