INTO CORRESPONDING FIELDS OF TABLE gt_itab
  FROM sflight
  WHERE carrid = zpcarrid
  ORDER BY carrid connid fldate.
" 만약 실습예제가 Seat 기준으로 정렬하라했으면 아래에 sort문으로 추가해야함

LOOP AT gt_itab INTO gs_itab.
  gs_itab-tot_seatsmax = gs_itab-seatsmax + gs_itab-seatsmax_b + gs_itab-seatsmax_f.
  gs_itab-tot_seatsocc = gs_itab-seatsocc + gs_itab-seatsocc_b + gs_itab-seatsocc_f.
  MODIFY gt_itab FROM gs_itab.

  IF gs_itab-tot_seatsocc >= 300.
    WRITE / icon_green_light AS ICON.
  ELSEIF gs_itab-tot_seatsocc < 300 AND
         gs_itab-tot_seatsocc > 100.
    WRITE / icon_yellow_light AS ICON.
  ELSE.
    WRITE / icon_red_light AS ICON.

  ENDIF.

  WRITE: / TEXT-000, gs_itab-carrid,
         / TEXT-001, gs_itab-connid,
         / TEXT-002, gs_itab-fldate,
         / TEXT-003, | { gs_itab-seatsmax + gs_itab-seatsmax_b + gs_itab-seatsmax_f } |,
         / TEXT-004, | { gs_itab-seatsocc + gs_itab-seatsocc_b + gs_itab-seatsocc_f } |.
  ULINE.
ENDLOOP.
