FORM select_data .
  SELECT A~MB_ST "ZTMEMBER_B25
         A~MB_ID
         A~MB_MM

         B~RENT_DT "ZTRENT_B25
         B~RPLAN_DT
         B~RETURN_ST

         C~BOOK_ID "ZTBOOK_B25
         C~BOOK_NM

"JOIN 조건
    INTO CORRESPONDING FIELDS OF TABLE GT_DATA
    FROM ZTMEMBER_B25     AS A
    INNER JOIN ZTRENT_B25 AS B ON A~MB_ID EQ B~MB_ID
    INNER JOIN ZTBOOK_B25 AS C ON C~BOOK_ID EQ B~BOOK_ID

    WHERE A~MB_ID IN SO_MID.

ENDFORM.

    IF GS_DATA-RETURN_ST EQ 'A'.

    ELSEIF GS_DATA-RETURN_ST EQ 'A'

    LOOP AT GT_DATA INTO GS_DATA.
