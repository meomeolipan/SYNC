*&           change field value during debugging
*&---------------------------------------------------------------------*
REPORT zbc400_ded_debug_exercise.


CONSTANTS gc_max TYPE i VALUE 5.

DATA: gv_count TYPE i VALUE 20,
      gv_idx_inner TYPE i,
      gv_idx_outer TYPE i.


*Part 2
************************************************************
WRITE 'Part 2:'.
ULINE.

WHILE gv_count > sy-index.

  gv_count = gv_count - sy-index.

  NEW-LINE.
  WRITE: 'GV_COUNT', gv_count.

ENDWHILE.

* Part 3
***********************************************************
SKIP.
WRITE 'Part 3:'.
ULINE.

DO gc_max TIMES.
  gv_idx_outer = sy-index.

  NEW-LINE.
  DO.
    gv_idx_inner = sy-index.

    IF gv_idx_inner < gv_idx_outer.
      WRITE gv_idx_inner.
    ELSEIF gv_idx_inner <= gc_max.
      WRITE gv_idx_outer.
    ELSE.
      EXIT.
    ENDIF.

  ENDDO.
ENDDO.

* Part 4
***********************************************************
SKIP.
WRITE 'Part 4:'.
ULINE.

IF gv_idx_inner <> gc_max.

*  RETURN.               " better use RETURN or STOP here
*  STOP.                " to avoid confusion regarding
*  EXIT.                " the different meanings of EXIT
*                       " inside and outside of loops
ENDIF.

SKIP.
WRITE: / 'User',
         sy-uname,
