MODULE GET_MEMNAME OUTPUT.
  IF GS_ZTMEMBER IS NOT INITIAL.
    MOVE-CORRESPONDING GS_ZTMEMBER TO ZTMEMBER_B09.
    IF ZTMEMBER_B09-MB_ST = 'A'.
      CALL FUNCTION 'ICON_CREATE'
        EXPORTING
          NAME   = ICON_GREEN_LIGHT
        IMPORTING
          RESULT = MY_ICON1.
    ELSEIF ZTMEMBER_B09-MB_ST = 'B'.
      CALL FUNCTION 'ICON_CREATE'
        EXPORTING
          NAME   = ICON_RED_LIGHT
        IMPORTING
          RESULT = MY_ICON1.
    ENDIF.

*  CHECK SY-SUBRC <> 0.
*  CLEAR GS_ZTMEMBER.

    ENDIF.
ENDMODULE.

MODULE FILL_DYNNR OUTPUT.

MODULE GET_BOOKINFO OUTPUT.
  IF GS_ZTBOOK IS NOT INITIAL.
    MOVE-CORRESPONDING GS_ZTBOOK TO ZTBOOK_B09.
    ZTRENT_B09-RPLAN_DT = SY-DATUM + 14.

    MOVE-CORRESPONDING GS_ZTMEMBER TO ZTRENT_B09.
    MOVE-CORRESPONDING GS_ZTBOOK TO ZTRENT_B09.
    MOVE 'A' TO ZTRENT_B09-RETURN_ST.
    MOVE SY-UNAME TO ZTRENT_B09-CHANGED_BY.
    MOVE SY-DATUM TO ZTRENT_B09-CHANGED_ON.
    MOVE SY-UZEIT TO ZTRENT_B09-CHANGED_TIME.

    MOVE-CORRESPONDING ZTRENT_B09 TO GS_ZTRENT.

*    CHECK SY-SUBRC <> 0.

  ENDIF.

*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
MODULE MODIFY_SCREEN OUTPUT.
    LOOP AT SCREEN.
      IF GS_ZTMEMBER-MB_ST = 'A' AND GS_ZTBOOK-BOOK_ST = 'A'.
        IF SCREEN-NAME = 'BTN_RENT'.
          SCREEN-INPUT = 1.
          MODIFY SCREEN.
        ENDIF.
      ELSE.
        IF SCREEN-NAME = 'BTN_RENT'.
          SCREEN-INPUT = 0.
          MODIFY SCREEN.
        ENDIF.
      ENDIF.
    ENDLOOP.
*    BREAK-POINT.

ENDMODULE.
