MODULE USER_COMMAND_0100 INPUT.

  CASE OK_CODE.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'FC1' OR 'FC2'.
      MY_TABSTRIP-ACTIVETAB = OK_CODE.
    WHEN 'BTNFC1'.
      DATA: LS_ZTRENT_B09 TYPE ZTRENT_B09.

      MOVE-CORRESPONDING GS_ZTRENT TO LS_ZTRENT_B09.
      INSERT ZTRENT_B09 FROM LS_ZTRENT_B09.

      IF SY-SUBRC = 0.
        MESSAGE TEXT-S01 TYPE 'S'.
      ELSE.
        MESSAGE TEXT-E01 TYPE 'W'.
      ENDIF.
      CLEAR GS_ZTBOOK.
      CLEAR GS_ZTMEMBER.
      CLEAR GS_ZTRENT.

  ENDCASE.

ENDMODULE.
