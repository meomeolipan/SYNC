FORM SELECT_DATA .

  "도서 대여 현황 데이트
  SELECT A~MB_ID A~MB_NM
         B~BOOK_ST B~BOOK_ID B~BOOK_NM
         C~RENT_DT C~RETURN_DT C~RETURN_ST
     INTO CORRESPONDING FIELDS OF TABLE GT_DATA
           FROM ZTRENT_B01 AS C
           INNER JOIN ZTBOOK_B01 AS B ON C~BOOK_ID = B~BOOK_ID
           INNER JOIN ZTMEMBER_B01 AS A ON C~MB_ID = A~MB_ID

       WHERE A~MB_ID IN SO_CAR
         AND C~RETURN_ST EQ PA_RST.



ENDFORM.

FORM SPLIT_CONTAINER .

  IF GO_DOCK1 IS NOT INITIAL.

    CREATE OBJECT GO_DOCK1
      EXPORTING
        REPID     = SY-REPID
        DYNNR     = SY-DYNNR
        SIDE      = GO_DOCK1->DOCK_AT_TOP
        EXTENSION = 1000.

    CREATE OBJECT GO_SPLIT
      EXPORTING
        PARENT  = GO_DOCK1
        ROWS    = 2
        COLUMNS = 1.

    "  split container에 container에 올리기.

    CALL METHOD GO_SPLIT->GET_CONTAINER
      EXPORTING
        ROW       = 1
        COLUMN    = 1
      RECEIVING
        CONTAINER = GO_CONT1.

    CALL METHOD GO_SPLIT->GET_CONTAINER
      EXPORTING
        ROW       = 2
        COLUMN    = 1
      RECEIVING
        CONTAINER = GO_CONT2.

    CALL METHOD GO_SPLIT->SET_ROW_HEIGHT
      EXPORTING
        ID     = 1
        HEIGHT = 10.

    PERFORM CREATE_CONT_GRID.


  ENDIF.

ENDFORM.
