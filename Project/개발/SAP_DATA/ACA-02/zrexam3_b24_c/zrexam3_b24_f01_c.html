FORM SELECT_DATA .
  SELECT A~MB_ID B~MB_NM A~BOOK_ID C~BOOK_NM
         A~RENT_DT A~RPLAN_DT A~RETURN_ST A~RETURN_DT
    INTO CORRESPONDING FIELDS OF TABLE GT_RENT
    FROM ZTRENT_B24 AS A
      INNER JOIN ZTMEMBER_B24 AS B ON A~MB_ID EQ B~MB_ID
      INNER JOIN ZTBOOK_B24 AS C ON A~BOOK_ID EQ C~BOOK_ID
    WHERE A~MB_ID IN SO_ID
      AND B~MB_ST EQ PA_ST
      AND A~RPLAN_DT IN SO_DT.
ENDFORM.

FORM MAKE_DATA .
  LOOP AT GT_RENT INTO GS_RENT.
    IF GS_RENT-RPLAN_DT < SY-DATUM AND GS_RENT-RETURN_ST ='A'.
      GS_RENT-LIGHT = '1'.
    ELSEIF GS_RENT-RPLAN_DT > SY-DATUM AND GS_RENT-RETURN_ST ='A'.
      GS_RENT-LIGHT = '2'.
    ELSEIF GS_RENT-RETURN_ST ='B'.
      GS_RENT-LIGHT = '3'.
    ENDIF.

    IF GS_RENT-RETURN_ST = 'A'.
      GS_RENT-RETURN_ST_TXT = '대여중'.
    ELSEIF GS_RENT-RETURN_ST = 'B'.
      GS_RENT-RETURN_ST_TXT = '반납완료'.
    ELSEIF GS_RENT-RETURN_ST = 'F'.
      GS_RENT-RETURN_ST_TXT = '연체'.
    ENDIF.

    " 펑션 써서 다시!!!!

    MODIFY GT_RENT FROM GS_RENT.
  ENDLOOP.

  SORT GT_RENT BY MB_ID BOOK_ID.
ENDFORM.
