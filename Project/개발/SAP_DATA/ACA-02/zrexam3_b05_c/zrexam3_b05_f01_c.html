*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM select_data .

  REFRESH gt_data.

  SELECT a~mb_id a~mb_nm
         b~book_id b~book_nm
         c~rent_dt c~rplan_dt c~return_st c~return_dt
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    FROM ztrent_b05   AS c
    JOIN ztmember_b05 AS a ON c~mb_id = a~mb_id
    JOIN ztbook_b05   AS b ON c~book_id = b~book_id
    WHERE a~mb_id     IN so_id
      AND a~mb_st     EQ pa_st
      AND c~rplan_dt  IN so_rp
    ORDER BY a~mb_id b~book_id.


ENDFORM.

FORM make_fieldcat_0100 .

  LOOP AT gt_fcat INTO gs_fcat.

    CASE gs_fcat-fieldname.
      WHEN 'LIGHT'.
        gs_fcat-coltext = '상태'.
      WHEN 'MB-ID'.
        gs_fcat-outputlen = 10.
        gs_fcat-just = 'L'.
        gs_fcat-key = 'X'.
      WHEN 'MB_NM'.
        gs_fcat-outputlen = 10.
        gs_fcat-just = 'L'.
        gs_fcat-key = 'X'.
      WHEN 'BOOK_ID'.
        gs_fcat-outputlen = 10.
        gs_fcat-just = 'L'.
      WHEN 'BOOK_NM'.
        gs_fcat-outputlen = 20.
        gs_fcat-just = 'L'.
      WHEN 'rent_dt'.
        gs_fcat-outputlen = 10.
        gs_fcat-just = 'L'.
      WHEN 'rplan_dt'.
        gs_fcat-outputlen = 10.
        gs_fcat-just = 'L'.
      WHEN 'return_st'.
        gs_fcat-outputlen = 10.
        gs_fcat-just = 'L'.
      WHEN 'return_dt'.
        gs_fcat-outputlen = 10.
        gs_fcat-just = 'L'.
    ENDCASE.

    MODIFY gt_fcat FROM gs_fcat.

  ENDLOOP.

ENDFORM.
