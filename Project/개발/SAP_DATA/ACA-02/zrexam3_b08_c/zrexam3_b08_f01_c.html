FORM GET_DATA1.

SELECT A~MB_ID
        A~MB_NM
        B~BOOK_ID
        B~BOOK_NM
        C~RENT_DT
        C~RPLAN_DT
        C~RETURN_ST
        C~RETURN_DT
  INTO CORRESPONDING FIELDS OF TABLE GT_DATA1
  FROM ZTMEMBER_B08 AS A
  INNER JOIN ZTRENT_B08 AS C
  ON A~MB_ID = C~MB_ID
  INNER JOIN ZTBOOK_B08 AS B
  ON B~BOOK_ID = C~BOOK_ID
  WHERE A~MB_ST = PA_MST
  AND A~MB_ID IN SO_MID
  AND C~RPLAN_DT IN SO_RPD.

  SORT GT_DATA1 BY MB_ID BOOK_ID.

ENDFORM.

FORM MAKE_DATA1.

  LOOP AT GT_DATA1 INTO GS_DATA1.

  IF GS_DATA1-RETURN_ST = 'A'.
      IF GS_DATA1-RPLAN_DT < SY-DATUM.
        GS_DATA1-LIGHT = '1'.
      ELSEIF GS_DATA1-RPLAN_DT > SY-DATUM.
        GS_DATA1-LIGHT = '2'.
      ENDIF.
  ELSEIF GS_DATA1-RETURN_ST = 'B'.
      GS_DATA1-LIGHT = '3'.
  ENDIF.

  MODIFY GT_DATA1
    FROM GS_DATA1
    TRANSPORTING LIGHT.

  ENDLOOP.

ENDFORM.

FORM USER_COMMAND USING P_E_UCOMM TYPE SY-UCOMM.

DATA : LT_ROW_NO TYPE LVC_T_ROID,
       LS_ROW_NO TYPE LVC_S_ROID.

DATA: LV_ANSWER TYPE STRING.

CALL METHOD GO_GRID3->GET_SELECTED_ROWS
    IMPORTING
      ET_ROW_NO     = LT_ROW_NO.

  IF LINES( LT_ROW_NO ) < 0.
    MESSAGE S014 DISPLAY LIKE 'E'.
  ELSEIF LINES( LT_ROW_NO ) > 1.
    MESSAGE S015 DISPLAY LIKE 'E'.

ENDIF.

IF SY-SUBRC = 0.
  IF P_E_UCOMM = 'RETURN'.
    IF LINES( LT_ROW_NO ) < 0.
      MESSAGE S014 DISPLAY LIKE 'E'.
    ELSEIF LINES( LT_ROW_NO ) > 1.
      MESSAGE S015 DISPLAY LIKE 'E'.
    ELSE.
      CALL FUNCTION 'POPUP_TO_CONFIRM'
        EXPORTING
         TITLEBAR                    = '반납처리확인'
         TEXT_QUESTION               = '해당 대여건을 반납 처리하시겠습니까?'
         TEXT_BUTTON_1               = '네'
         TEXT_BUTTON_2               = '아니요'
         DISPLAY_CANCEL_BUTTON       = ''
       IMPORTING
         ANSWER                      = LV_ANSWER.

      IF SY-SUBRC <> 0.
        MESSAGE S013 DISPLAY LIKE 'E'.
      ENDIF.

      IF LV_ANSWER = 1.
        UPDATE ZTRENT_B08
        SET RETURN_ST = 'B'
        RETURN_DT = SY-DATUM
        CHANGED_BY = SY-UNAME
        CHANGED_ON = SY-DATUM
        CHANGED_TIME = SY-UZEIT
        WHERE BOOK_ID = GS_DATA3-BOOK_ID.

        IF SY-SUBRC = 0.
          MESSAGE S012.
        ELSE.
          MESSAGE S013 DISPLAY LIKE 'E'.
        ENDIF.
       ENDIF.
    ENDIF.

  ENDIF.
  PERFORM REFRESH_ALV_GRID.
ENDIF.

ENDFORM.
