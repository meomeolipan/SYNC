  LOOP AT SCREEN.

      IF SCREEN-GROUP1 = 'MO1'.

      ENDIF.
    ELSE.
      IF SCREEN-GROUP1 = 'MO1'.
        SCREEN-ACTIVE = GV_SWITCH_OFF.
      ENDIF.
    ENDIF.
    IF PA_2 = 'X'.
      IF SCREEN-GROUP1 = 'MO2'.
        SCREEN-ACTIVE = GV_SWITCH.
      ENDIF.
    ELSE.
      IF SCREEN-GROUP1 = 'MO2'.
        SCREEN-ACTIVE = GV_SWITCH_OFF.
      ENDIF.
    ENDIF.
    MODIFY SCREEN.
  ENDLOOP.

START-OF-SELECTION.

  PERFORM DISPLAY_DATA.
