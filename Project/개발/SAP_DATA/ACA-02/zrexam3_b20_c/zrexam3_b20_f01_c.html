FORM get_rent_data .
  CLEAR gt_rent.

  SELECT a~mb_id a~mb_nm
        b~book_id b~book_nm
        c~rent_dt c~rplan_dt c~return_st c~return_dt
    INTO CORRESPONDING FIELDS OF TABLE gt_rent
        FROM ztrent_b20 AS c
  INNER JOIN ztbook_b20 AS b ON b~book_id = c~book_id
  INNER JOIN ztmember_b20 AS a ON a~mb_id = c~mb_id
       WHERE c~mb_id    IN so_mbid
         AND a~mb_st    = pa_mbst
         AND c~rplan_dt IN so_rpdt.


ENDFORM.
