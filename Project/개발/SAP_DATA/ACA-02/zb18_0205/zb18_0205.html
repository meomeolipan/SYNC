**********************************************************************
* SCARR 테이블의 형식과 같게
**********************************************************************

* 1. dictionary에 있는 structure 으로 structured type 변수를 선언한다.
" 1) Table (SCARR)
DATA: gs_scarr_a TYPE scarr.
" 2) STRUCTURE 생성 (SCARR 테이블과 동일하게 : ZSSCARR_B##)
DATA: gs_scarr_b TYPE zsscarr_b18.
* 2. TYPES 선언으로 structured type 변수 선언
TYPES : BEGIN OF ty_scarr,
          carrid   TYPE s_carr_id,
          carrname TYPE s_carrname,
          currcode TYPE s_currcode,
          url      TYPE s_carrurl,
        END OF ty_scarr.

DATA: gs_scarr_c TYPE ty_scarr.
* 3. begin of ~ 활용해서 직접 local 하게 structured type 변수 선언
DATA: BEGIN OF gs_scarr_d,
        carrid   TYPE s_carr_id,
        carrname TYPE s_carrname,
        currcode TYPE s_currcode,
        url      TYPE s_carrurl,
      END OF gs_scarr_d.

* 값 assign
MOVE 'aa' TO gs_scarr_a-carrid.
MOVE 'american airline' TO gs_scarr_a-carrname.
gs_scarr_a-currcode = 'USD'.
gs_scarr_a-url = 'www.naver.com'.

* MOVE-CORRESPONDING gs_scarr_a TO gs_scarr_b.
gs_scarr_b-carrid = gs_scarr_a-carrid.
gs_scarr_b-carrname = gs_scarr_a-carrname.
gs_scarr_b-currcode = gs_scarr_a-currcode.
gs_scarr_b-url = gs_scarr_a-url.

MOVE-CORRESPONDING gs_scarr_b TO gs_scarr_c.
MOVE-CORRESPONDING gs_scarr_b TO gs_scarr_d.

WRITE :/ gs_scarr_d-carrid, gs_scarr_d-carrname, gs_scarr_d-currcode, gs_scarr_d-url(20), "괄호 안에 쓴 숫자만큼만 공간 확보
       / gs_scarr_a. " 구조체를 Wirte문에 쓰면 알아서 멤버들을 전부 출력해준다.

*--------------------------------------------------------------------*
* internal table에 값 입력하고 출력해보기
*--------------------------------------------------------------------*

DATA :gt_scarr TYPE TABLE OF scarr,
      gs_scarr LIKE LINE OF gt_scarr.

gs_scarr-carrid = 'AA'.
gs_scarr-carrname = 'Seoul Airport'.
gs_scarr-currcode = 'KRW'.
gs_scarr-url = 'www.naver.com'.

" INSERT [wa INTO | INITIAL LINE INTO] itab [INDEX idx]. standard와 sorted에만 사용 가능
" INSERT [wa INTO INITIAL LINE INTO] TABLE itab. 모든 테이블에 사용 가능

INSERT gs_scarr INTO TABLE gt_scarr.
CLEAR gs_scarr.

gs_scarr-carrid = 'AB'.
gs_scarr-carrname = 'LA Airport'.
gs_scarr-currcode = 'USD'.
gs_scarr-url = 'www.google.com'.

INSERT gs_scarr INTO gt_scarr INDEX 1. " 1번 index에 삽입한다.
" INSERT gs_scarr INTO table gt_scarr index 1. 에러
CLEAR gs_scarr.

gs_scarr-carrid = 'AC'.
gs_scarr-carrname = 'Africa'.
gs_scarr-currcode = 'USD'.
gs_scarr-url = 'www.Africa.com'.

APPEND gs_scarr TO gt_scarr. " 테이블 끝에 데이터를 삽입한다.
CLEAR gs_scarr.

LOOP AT gt_scarr INTO gs_scarr.
  WRITE :/ gs_scarr.
ENDLOOP.

READ TABLE gt_scarr into gs_scarr WITH KEY carrid = 'AC'.
