MODULE modify_screen INPUT.

  LOOP AT SCREEN.
    IF screen-name = ''.
      screen-input = 1.
    ENDIF.
    IF screen-name = ' ' .
      screen-input = 0.
    ENDIF.
    MODIFY SCREEN.
  ENDLOOP.
ENDMODULE.
