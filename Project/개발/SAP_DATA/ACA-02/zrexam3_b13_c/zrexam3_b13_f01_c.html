FORM MODIFY_DATA .
  CLEAR GS_RENTST.

    IF GS_RENTST-RETURN_ST = 'A'.
      GS_RENTST-RETURN_ST_C = '대여중'.

      GS_RENTST-RETURN_ST_C = '반납완료'.
    ELSEIF GS_RENTST-RETURN_ST = 'F'.

    ENDIF.

    "대여중이면서 연체
    IF GS_RENTST-RPLAN_DT < SY-DATUM AND GS_RENTST-RETURN_ST = 'A'.
      GS_RENTST-LIGHT = '1'. "빨강

      "대여중이면서 연체가 아닌 것
    ELSEIF GS_RENTST-RPLAN_DT >= SY-DATUM AND GS_RENTST-RETURN_ST = 'A'.
      GS_RENTST-LIGHT = '2'. "노랑
      "반납완료
    ELSEIF GS_RENTST-RETURN_ST = 'B'.
      GS_RENTST-LIGHT = '3'. "초록
    ENDIF.

    MODIFY GT_RENTST FROM GS_RENTST.
    CLEAR GS_RENTST.

  ENDLOOP.

  "출력 데이터의 정렬
