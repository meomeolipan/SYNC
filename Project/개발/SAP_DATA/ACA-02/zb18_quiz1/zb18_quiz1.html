REPORT zb18_quiz1.

TABLES : vbak, tvakt.

DATA : BEGIN OF gs_vbak,
         vbeln TYPE vbak-vbeln,
         erdat TYPE vbak-erdat,
         trvog TYPE vbak-trvog,
         vbtyp TYPE vbak-vbtyp,
         auart TYPE vbak-auart,
         bezei TYPE tvakt-bezei,
         desc  TYPE c LENGTH 50,
       END OF gs_vbak.

DATA : gt_vbak LIKE TABLE OF gs_vbak.

SELECT * FROM vbak INTO CORRESPONDING FIELDS OF TABLE gt_vbak.

LOOP AT gt_vbak INTO gs_vbak.
*--------------------------------------------------------------------*
* 현재 구조체에서 bezei와 desc가 비어있는데 아래 두 function을 실행하여 하나씩 채워준다.
*--------------------------------------------------------------------*

  call function 'ZFMB18_GET_BEZEI'"텍스트 테이블에서 auart와 로그인 언어가 일치하는 데이터를 검색하여 텍스트를 반환
    EXPORTING
      iv_auart = gs_vbak-auart
    IMPORTING
      ev_bezei = gs_vbak-bezei.

  call function 'ZFMB18_GET_DESC'"vbtyp의 값(A, B, C)에 따라 텍스트를 반환
    EXPORTING
      iv_vbtyp = gs_vbak-vbtyp
    IMPORTING
      ev_desc  = gs_vbak-desc.

  MODIFY gt_vbak FROM gs_vbak TRANSPORTING bezei desc.

*--------------------------------------------------------------------*
* 출력부
*--------------------------------------------------------------------*
  WRITE : / gs_vbak-vbeln,
            gs_vbak-erdat,
            gs_vbak-trvog,
            gs_vbak-vbtyp,
            gs_vbak-auart,
