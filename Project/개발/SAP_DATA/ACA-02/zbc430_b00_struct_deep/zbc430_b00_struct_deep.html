*&---------------------------------------------------------------------*
*& Report ZBC430_B00_STRUCT_DEEP
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
REPORT zbc430_b00_struct_deep.


DATA gs_person TYPE zperson_b00. "structure
DATA gs_phone TYPE str_phone. "work area


START-OF-SELECTION.

*Fill deep structure with data
  gs_person-name-firstname = 'Harry'.
  gs_person-name-lastname = 'potter'.

  gs_person-street = 'Privet Drive'.
  gs_person-nr = '3'.
  gs_person-zip = 'GB-10889'.
  gs_person-city = 'London'.

  " 새로 추가된 테이블 타입(=인터널 테이블)에 데이터를 넣는 방법

*1. structure의 component 가 가진 table type의 line type형태의 work area에 입력
  gs_phone-p_type = 'P'.
  gs_phone-p_number = '+31-10-9938990'.

*2. 구성된 work area의 값을 table type에 넣는다.
  INSERT gs_phone INTO TABLE gs_person-phone.

  gs_phone-p_type = 'F'.
  gs_phone-p_number = '+31-10-9938991'.
  INSERT gs_phone INTO TABLE gs_person-phone.

  gs_phone-p_type = 'M'.
  gs_phone-p_number = '+31-79-12211433'.
  INSERT gs_phone INTO TABLE gs_person-phone.


*Write on List
  WRITE: /  gs_person-name-firstname ,
            gs_person-name-lastname ,
            gs_person-street ,
            gs_person-nr ,
            gs_person-zip ,
            gs_person-city .

  WRITE: / 'Phone-Numbers:'.

  LOOP AT gs_person-phone INTO gs_phone.
    WRITE:  AT 20 gs_phone-p_type,
                  gs_phone-p_number.
    NEW-LINE.
  ENDLOOP.
