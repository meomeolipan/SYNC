MODULE fill_dynnr OUTPUT.

  CASE ok_code.
    WHEN 'FC1'.
      dynnr = 110.
    WHEN 'FC2'.
      dynnr = 120.
    WHEN OTHERS.
      tab-activetab = 'FC1'.
      dynnr = 110.
  ENDCASE.

ENDMODULE.

MODULE set_member_st OUTPUT.

  CASE ztmember-mb_st.
    WHEN 'A'.
      icon = icon_green_light.
    WHEN 'B'.
      icon = icon_red_light.
    WHEN OTHERS.
  ENDCASE.

ENDMODULE.

*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
MODULE set_icon_st OUTPUT.

  IF ztmember-mb_st = 'A' AND ztbook-book_st = 'A'.
  ELSE.
    LOOP AT SCREEN.
      IF screen-name = 'PUSH'.
        screen-input = 0.
        MODIFY SCREEN.
      ENDIF.
    ENDLOOP.
  ENDIF.
ENDMODULE.
