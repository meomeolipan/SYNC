*&---------------------------------------------------------------------*
*& INITIALIZATION
*&---------------------------------------------------------------------*
INITIALIZATION.
  CLEAR : SO_RPL, SO_RPL[].
  SO_RPL-SIGN = 'I'.
  SO_RPL-OPTION = 'BT'.
  SO_RPL-LOW = '20240301'.
  SO_RPL-HIGH = SY-DATUM.
  APPEND SO_RPL.


*&---------------------------------------------------------------------*
*& AT SELECTION-SCREEN OUTPUT.
*&---------------------------------------------------------------------*
AT SELECTION-SCREEN OUTPUT.

  CASE GV_SWITCH.
    WHEN '1'.
      LOOP AT SCREEN.
        IF SCREEN-GROUP1 = 'M1'.
          SCREEN-ACTIVE = '1'.
        ENDIF.
        IF SCREEN-GROUP1 = 'M2'.
          SCREEN-ACTIVE = '0'.
        ENDIF.
        MODIFY SCREEN.
      ENDLOOP.
    WHEN OTHERS.
      LOOP AT SCREEN.
        IF SCREEN-GROUP1 = 'M1'.
          SCREEN-ACTIVE = '0'.
        ENDIF.
        IF SCREEN-GROUP1 = 'M2'.
          SCREEN-ACTIVE = '1'.
        ENDIF.
        MODIFY SCREEN.
      ENDLOOP.
  ENDCASE.



*&---------------------------------------------------------------------*
*& AT SELECTION-SCREEN
*&---------------------------------------------------------------------*
AT SELECTION-SCREEN.
  CASE SSCRFIELDS-UCOMM.
    WHEN 'CLK'.
      IF GV_SWITCH = '1'.
        GV_SWITCH = '0'.
      ELSE.
        GV_SWITCH = '1'.
      ENDIF.
    WHEN OTHERS.
  ENDCASE.

*&---------------------------------------------------------------------*
*& START-OF-SELECTION
*&---------------------------------------------------------------------*
START-OF-SELECTION.

  CASE GV_SWITCH.
    WHEN '1'.
      PERFORM GET_DATA1.
      PERFORM MAKE_DATA1.
      PERFORM DISPLAY_DATA1.
    WHEN '0'.
      PERFORM GET_DATA2_1.
      PERFORM MAKE_DATA2_1.
      PERFORM DISPLAY_DATA2_1.
  ENDCASE.
*&---------------------------------------------------------------------*
*& Form GET_DATA2_2
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM GET_DATA2_2 .
  SELECT * INTO CORRESPONDING FIELDS OF TABLE GT_DISPLAY2_2
    FROM ZTRENT_B09 AS A
    INNER JOIN ZTBOOK_B09 AS B
    ON A~BOOK_ID = B~BOOK_ID
    WHERE MB_ID = GS_DISPLAY2_1-MB_ID
    AND RETURN_ST EQ 'A'.
ENDFORM.
