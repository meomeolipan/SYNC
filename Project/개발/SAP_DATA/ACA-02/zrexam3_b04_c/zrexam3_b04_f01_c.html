FORM select_data .

  CLEAR gt_data.

  SELECT a~mb_id a~mb_nm
         b~book_id b~book_nm
         c~rent_dt c~rplan_dt c~return_st c~return_dt
    INTO CORRESPONDING FIELDS OF TABLE gt_data
    FROM ztrent_b04 AS c
    INNER JOIN ztmember_b04 AS a ON a~mb_id EQ c~mb_id
    INNER JOIN ztbook_b04 AS b ON b~book_id EQ c~book_id

     WHERE a~mb_id     IN s_id
      AND a~mb_st      EQ s_st
      AND c~rplan_dt   IN s_dt.

ENDFORM.

*& Form MODIFY_DATA
*&---------------------------------------------------------------------*

*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM modify_data .

  CLEAR gt_display.

  LOOP AT gt_data INTO gs_data.
    CLEAR gs_display.
    MOVE-CORRESPONDING gs_data TO gs_display.

    IF gs_display-rplan_dt GT sy-datum
      AND gs_display-return_st EQ 'B'.
      gs_display-light = '1'. "빨간색

    ELSEIF gs_display-rplan_dt LE sy-datum
      AND gs_display-return_st EQ 'B'.
      gs_display-light = '2'. "노란색
    ELSE.
      gs_display-light = '3'. "초록색
    ENDIF.

    APPEND gs_display TO gt_display.
  ENDLOOP.

ENDFORM.

FORM double_click USING Pv_ROW TYPE lvc_s_row
                            Pv_COLUMN TYPE lvc_s_col
                            Pv_ROW_NO TYPE lvc_s_roid.

  CLEAR gs_data.
  READ TABLE gt_data INTO gs_data INDEX Pv_ROW-index.

  SELECT b~book_id b~book_nm
         c~rent_dt c~rplan_dt
     INTO CORRESPONDING FIELDS OF TABLE gt_display
     FROM ztrent_b04 AS c
     INNER JOIN ztmember_b04 AS a ON a~mb_id EQ c~mb_id
     INNER JOIN ztbook_b04 AS b ON b~book_id EQ c~book_id.


  CLEAR GS_display.
  LOOP AT GT_display INTO GS_display.

    MODIFY GT_display FROM GS_display.
    CLEAR GS_display.

  ENDLOOP.

  PERFORM set_fcat.
  PERFORM set_layo.
  PERFORM call_alv.
