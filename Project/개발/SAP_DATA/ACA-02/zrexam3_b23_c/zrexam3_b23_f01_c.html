    b~book_id
    b~book_nm

    c~return_st
    c~return_dt


    INTO CORRESPONDING FIELDS OF TABLE gt_rent
    FROM ztrent_b23 AS c INNER JOIN ztmember_b23 AS a

    ON c~mb_id = a~mb_id

    INNER JOIN ztbook_b23 AS b
    ON c~book_id = b~book_id

    WHERE a~mb_id IN so_mid
      AND a~mb_st EQ so_mst
