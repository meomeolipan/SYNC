FORM get_data .
  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE gt_rent
      FROM ztrent_b21
      INNER JOIN ztmember_b21
      ON ztrent_b21~mb_id = ztmember_b21~mb_id
      INNER JOIN ztbook_b21
      ON ztrent_b21~book_id = ztbook_b21~book_id
      WHERE ztrent_b21~mb_id IN p_id
      AND ztmember_b21~mb_st = p_stu
      AND ztrent_b21~rplan_dt IN p_rd
      ORDER BY ztrent_b21~mb_id ztrent_b21~book_id.
ENDFORM.
