REPORT ZB18_TEST07.

TYPES: BEGIN OF t_sflight,
  carrid TYPE s_carr_id,
  connid TYPE s_conn_id,
  seatsmax type S_SEATSMAX,
  seatsocc type s_seatsocc,
  paymentsum type s_sum,
 END OF t_sflight.

DATA: it_sflight TYPE table of t_sflight,
      wa_sflight TYPE t_sflight,
      it_sum type SORTED TABLE OF t_sflight with unique key carrid connid.

SELECT *
  FROM sflight
  INTO CORRESPONDING FIELDS OF TABLE it_sflight.

loop at it_sflight into wa_sflight.
    collect wa_sflight into it_sum.
  endloop.

  loop at it_sum into wa_sflight.
    write: / wa_sflight-carrid,
             wa_sflight-connid,
             wa_sflight-seatsmax,
             wa_sflight-seatsocc,
             wa_sflight-paymentsum.
  endloop.
