MODULE fill_dynnr OUTPUT.
  CASE tab-activetab.
    WHEN 'FC1'.
      dynnr = 120.
    WHEN 'FC2'.
      dynnr = 110.
    WHEN OTHERS. " 초기화면 설정

ENDMODULE.

ENDMODULE.
*&---------------------------------------------------------------------*
*& Module MODIFY_SCREEN OUTPUT
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
MODULE modify_screen OUTPUT.
  IF gs_book-book_st = 'A' AND gs_member-mb_st = 'A'.
    LOOP AT SCREEN.
      IF screen-name = 'P1'.
        screen-input = 1.
      ENDIF.
      MODIFY SCREEN.
    ENDLOOP.
  ELSE.
    LOOP AT SCREEN.
      IF screen-name = 'P1'.
        screen-input = 0.
      ENDIF.
      MODIFY SCREEN.
    ENDLOOP.
  ENDIF.
ENDMODULE.
