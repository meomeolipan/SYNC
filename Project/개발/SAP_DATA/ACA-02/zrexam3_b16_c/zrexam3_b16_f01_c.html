FORM modify_screen .

    CASE 'X'.

        IF screen-group1 = 'REN'.
          screen-active = '0'.
        ELSEIF screen-group1 = 'MID'.
          screen-active = '1'.
        ENDIF.
      WHEN pa_ren.
        IF screen-group1 = 'REN'.
          screen-active = '1'.
        ELSEIF screen-group1 = 'MID'.
          screen-active = '1'.
        ENDIF.
    ENDCASE.
    MODIFY SCREEN.
  ENDLOOP.
ENDFORM.

FORM get_data .

  SELECT  a~mb_id  a~mb_nm
          b~book_id
          c~book_nm
          b~rent_dt b~rplan_dt b~return_st b~return_dt
    INTO CORRESPONDING FIELDS OF TABLE gt_rent
    FROM ztmember_b16 AS a INNER JOIN ztrent_b16 AS b
    ON a~mb_id = b~mb_id
    INNER JOIN ztbook_b16 AS c
    ON b~book_id = c~book_id
  WHERE a~mb_id IN so_ID
    AND a~mb_st = pa_st
    AND b~rplan_dt IN so_ret
    ORDER BY a~mb_id b~book_id.


  SELECT * INTO CORRESPONDING FIELDS OF TABLE gt_return
    FROM ztmember_b16
    WHERE  mb_id IN so_ID.


ENDFORM.
