REPORT zbc405_b18_group_by.

TABLES sflight.

DATA : go_alv_grid TYPE REF TO cl_salv_table.

SELECT-OPTIONS so_car FOR sflight-carrid.

TYPES : BEGIN OF gty_sflight,
          carrid    TYPE sflight-carrid,
          connid    TYPE sflight-connid,
          seatsmax  TYPE sflight-seatsmax,
          seatsocc  TYPE sflight-seatsocc,
          searchnum TYPE i,
        END OF gty_sflight.

DATA : gt_sflight TYPE TABLE OF gty_sflight.

START-OF-SELECTION.



  "SELECT carrid connid SUM( seatsmax ) AS seatsmax SUM( seatsocc ) AS seatsocc COUNT( * ) AS searchnum
  select carrid connid sum( seatsmax ) sum( seatsocc ) count( * )
    FROM sflight
    "INTO CORRESPONDING FIELDS OF TABLE gt_sflight
    into table gt_sflight
    WHERE carrid IN so_car
    GROUP BY carrid connid.

    CALL METHOD cl_salv_table=>factory
    IMPORTING
      r_salv_table = go_alv_grid
    CHANGING
      t_table      = gt_sflight.

  go_alv_grid->display( ).
