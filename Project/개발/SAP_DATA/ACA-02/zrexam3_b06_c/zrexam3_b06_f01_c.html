FORM SELECT_DATA_0100 .
  CLEAR GT_DATA.
  SELECT B~MB_ID B~MB_NM C~BOOK_ID C~BOOK_NM
         A~RENT_DT A~RPLAN_DT A~RETURN_DT A~RETURN_ST
  INTO CORRESPONDING FIELDS OF TABLE GT_DATA
  FROM ZTRENT_B06   AS A    INNER JOIN
       ZTMEMBER_B06 AS B    ON A~MB_ID = B~MB_ID
                            INNER JOIN
       ZTBOOK_B06   AS C    ON A~BOOK_ID = C~BOOK_ID.

  PERFORM MODIFY_DATA.
ENDFORM.
