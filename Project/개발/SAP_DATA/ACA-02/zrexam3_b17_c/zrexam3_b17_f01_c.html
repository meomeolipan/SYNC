  LOOP AT gt_display1 INTO gs_display1.

        "대여중이면서 반납 예정일이 안지낫으면 -> ㄱㅊ (노)
      ELSEIF gs_display1-return_st = 'A' AND gs_display1-rplan_dt GE sy-datum.
        gs_display1-light = 2.
      ELSEIF gs_display1-return_st = 'B'. "반납완료 (초)
        gs_display1-light = 3.
      ENDIF.

      CLEAR gs_cell_style.
      gs_cell_style-fieldname = 'DISPLAY_STATUS'.
      CASE gs_display1-return_st.
        WHEN 'A'.

        WHEN 'B'. "반납완료

        WHEN 'F'. "연체
          gs_display1-display_status = '연체'.

*            OTHERS        = 1.
*        IF sy-subrc <> 0.
*        ENDIF.
*      ENDIF.

      MODIFY gt_display1 FROM gs_display1 TRANSPORTING light
                                                       display_status
                                                       it_cell_style.
    ENDLOOP.

  IF gt_display2 IS NOT INITIAL.
    LOOP AT gt_display2 INTO gs_display2.
*           연체료 추가
      CLEAR gs_cell_style.
      gs_cell_style-fieldname = 'DISPLAY_FEE'.
      gs_display2-display_fee = '10000'."( sy-datum - gs_display2-rplan_dt ) * 300.
