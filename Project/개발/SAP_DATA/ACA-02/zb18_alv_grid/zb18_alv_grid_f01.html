FORM get_data .
  CASE 'X'.
    WHEN pa_all.

        a~connid IN so_con AND
        a~fldate IN so_fld AND
        a~cancelled = pa_canc.

    WHEN pa_eco.

  LEFT OUTER JOIN scarr AS b

  WHERE a~carrid IN so_car AND
        a~connid IN so_con AND

FORM make_data .
  LOOP AT gt_sbook INTO gs_sbook.
    IF gs_sbook-custtype = 'B'.
      gs_sbook-light = '3'.
    ELSE.
      gs_sbook-light = '2'.
    ENDIF.

* Row Color 적용
    IF gs_sbook-smoker = 'X'.
      gs_sbook-color = 'C' && col_negative && '00'.
    ENDIF.

* Cell Color 적용
    IF gs_sbook-invoice = 'X'.
      gs_colfields-fname = 'INVOICE'.
      gs_colfields-color-col = col_positive.
      gs_colfields-color-int = 0.
      gs_colfields-color-inv = 1.
      APPEND gs_colfields TO gs_sbook-it_colfields.
    ENDIF.

* Icon 적용
    IF gs_sbook-cancelled = 'X'.
      gs_sbook-cancelled_icon = icon_red_light.
    ELSE.
      gs_sbook-cancelled_icon = icon_green_light.
    ENDIF.



    MODIFY gt_sbook FROM gs_sbook TRANSPORTING light Color it_colfields cancelled_icon.
  ENDLOOP.
ENDFORM.
