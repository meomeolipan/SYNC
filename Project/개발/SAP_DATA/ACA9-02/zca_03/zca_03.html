REPORT ZCA_03.

**********************************************************************
*변수 선언
**********************************************************************
DATA : LT_SFLIGHT TYPE TABLE OF SFLIGHT,
       LS_SFLIGHT TYPE SFLIGHT,
*       LS_SFLIGHT LIKE LINE OF LT_SFLIGHT,

       LT_SFLIGHT_HEADER TYPE TABLE OF SFLIGHT WITH HEADER LINE.

**********************************************************************
*쿼리
**********************************************************************
SELECT *
  INTO CORRESPONDING FIELDS OF TABLE LT_SFLIGHT
  UP TO 10 ROWS
  FROM SFLIGHT.

MOVE LT_SFLIGHT TO LT_SFLIGHT_HEADER[].

**********************************************************************
*헤더라인이 없는 인터널 테이블
**********************************************************************
LOOP AT LT_SFLIGHT INTO LS_SFLIGHT.

  IF LS_SFLIGHT-CARRID EQ 'AA'.
      LS_SFLIGHT-CONNID = '1000'.
  ENDIF.

  MODIFY LT_SFLIGHT FROM LS_SFLIGHT.
ENDLOOP.

**********************************************************************
*헤더라인이 있는 인터널 테이블
**********************************************************************
LOOP AT LT_SFLIGHT_HEADER.

  IF LS_SFLIGHT-CARRID EQ 'AA'.
      LS_SFLIGHT-CONNID = '1000'.
  ENDIF.

  MODIFY LT_SFLIGHT_HEADER.
ENDLOOP.
