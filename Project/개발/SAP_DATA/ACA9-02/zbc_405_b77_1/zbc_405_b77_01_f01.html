FORM MODIFY_SCREEN .

   CASE SY-DYNNR.
    WHEN 1100.
      LOOP AT SCREEN.
        IF SCREEN-GROUP1 = 'DET'.
          SCREEN-ACTIVE = GV_SWITCH. " 초기값은 0이다.
          MODIFY SCREEN.
        ENDIF.
      ENDLOOP.

      IF GV_SWITCH = '0'.
        GV_TXT = TEXT-P01.
      ELSE.
        GV_TXT = TEXT-P02.
*       clear additional select-options to avoid unwanted
*       influence at selection from database
        REFRESH: SO_START,
                 SO_DEST.
      ENDIF.
  ENDCASE.

ENDFORM.
