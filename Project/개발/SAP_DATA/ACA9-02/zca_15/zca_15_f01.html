FORM CREATE_OBJECT_PERSON USING PV_NAME TYPE FULLNAME
                                  PV_MONEY TYPE I.

  DATA LV_MSG TYPE STRING.

  CREATE OBJECT GO_PERSON
  EXPORTING
    IV_NAME    = PV_NAME
    IV_MONNEY  = PV_MONEY
  EXCEPTIONS
    EMPTY_NAME = 1
    OTHERS     = 2.

CASE SY-SUBRC.
  WHEN 0.
    APPEND GO_PERSON TO GT_PERSON.

    CONCATENATE  '사람' PV_NAME '이 생성됐다!!'
    INTO LV_MSG SEPARATED BY SPACE.

*    WRITE : / '사람', PV_NAME, '이 생성됐다!!' .
    WRITE : LV_MSG.
    ULINE.


  WHEN 1.
    WRITE : / '이름 없이 승객을 생성하고자 했습니다.'.
    ULINE.
  WHEN OTHERS.
    WRITE : / '알 수 없는 오류가 발생했습니다.'.
    ULINE.
ENDCASE.

ENDFORM.

FORM CREATE_OBJECT_BUS USING PV_NUMBER TYPE CHAR10
                             PV_FARE TYPE I.

  CREATE OBJECT GO_BUS
    EXPORTING
      IV_NUMBER       = PV_NUMBER
      IV_FARE         = PV_FARE
    EXCEPTIONS
      NO_INPUT_NUMBER = 1
      NO_INPUT_FARE   = 2
      OTHERS          = 3.

  CASE SY-SUBRC.
    WHEN 0.
      APPEND GO_BUS TO GT_BUS.
      WRITE : / '버스', PV_NUMBER, '이 생성됐다.'.
      ULINE.
    WHEN 1.
      WRITE : / '버스 노선번호 없이 생성 불가.'.
      ULINE.
    WHEN 2.
      WRITE : '버스', PV_NUMBER, '이 운임없이 생성하고자 했다'.
      ULINE.
    WHEN OTHERS.
      WRITE : / '알 수 없는 오류가 발생했습니다.'.
  ENDCASE.

ENDFORM.

FORM TAKE_BUS USING PV_PERSON_INDEX TYPE I
                    PV_BUS_INDEX    TYPE I.

READ TABLE GT_PERSON INTO GO_PERSON INDEX PV_PERSON_INDEX.
IF SY-SUBRC NE 0.
  WRITE : / '현재 생성된 승객들보다 큰 숫자를 입력 또는',
          / '1보다 작은 숫자를 입력했습니다.'.
  EXIT.
ENDIF.

READ TABLE GT_BUS INTO GO_BUS INDEX PV_BUS_INDEX.
IF SY-SUBRC NE 0.
  WRITE : / '현재 생성된 버스보다 큰 숫자를 입력 또는',
          / '1보다 작은 숫자를 입력했습니다.'.
  EXIT.
ENDIF.

DATA LV_FARE TYPE I.

LV_FARE = GO_BUS->GET_FARE( ).

*GO_PERSON->USE_MONEY( LV_FARE ).
  GO_PERSON->USE_MONEY(
    EXPORTING
      IV_MONEY = LV_FARE
    EXCEPTIONS
      INPUT_MONEY_MINUS = 1
      NOT_ENOUGH_MONEY  = 2
      OTHERS            = 3
  ).


CASE SY-SUBRC.
  WHEN 0.
    GO_BUS->ADD_PASSENGER( ).

  WHEN 1.

  WHEN 2.

  WHEN OTHERS.

ENDCASE.



ENDFORM.
