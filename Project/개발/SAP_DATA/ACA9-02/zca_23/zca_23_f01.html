FORM ENTER_CUSTOMER.

_INIT GT_BDCDATA GS_BDCDATA.
PERFORM DYNPRO USING : 'X' 'SAPBC402_PGCD_CREATE_CUSTOMER' '0100',
                       ' ' 'BDC_OKCODE'                    '=SAVE',
                       ' '  'SCUSTOM-NAME'                 PA_NAME,
                       ' '  'SCUSTOM-FORM'                 PA_FORM,
                       ' '  'SCUSTOM-STREET'               PA_STRT,
                       ' '  'SCUSTOM-POSTCODE'             PA_POST,
                       ' '  'SCUSTOM-CITY'                 PA_CITY,
                       ' '  'SCUSTOM-COUNTRY'                'KR',
                       ' '  'SCUSTOM-EMAIL'                PA_EMAIL.

*---SET BDC Option
**********************************************************************
*그냥 붙이기
**********************************************************************
GS_PARAMS-DISMODE = PA_MODE.
GS_PARAMS-UPDMODE = 'S'.
GS_PARAMS-NOBIEND = SPACE.
GS_PARAMS-RACOMMIT = 'X'.

**********************************************************************

_init GT_MSGCOLL GS_MSGCOLL.
 CALL TRANSACTION 'BC402_CALD_CRE_CUST' USING GT_BDCDATA
                                        OPTIONS FROM GS_PARAMS
                                        MESSAGES INTO GT_MSGCOLL.

**********************************************************************
* Success confirm
 CLEAR GT_MSGCOLL.
 READ TABLE GT_MSGCOLL INTO GS_MSGCOLL WITH KEY MSGID = 'BC402_C62'
                                                MSGNR = '015'.
 IF SY-SUBRC EQ 0.
   CLEAR GV_MSG.

   "매크로 사용
    _MSG_BUILD GS_MSGCOLL-MSGID GS_MSGCOLL-MSGNR
               GS_MSGCOLL-MSGV1 GS_MSGCOLL-MSGV2 GS_MSGCOLL-MSGV3 GS_MSGCOLL-MSGV4
               GV_MSG.

*   CALL FUNCTION 'MMESSAGE_TEXT_BUILD'
*     EXPORTING

*       MSGNR               =   GS_MSGCOLL-MSGNR               " Number of message
*       MSGV1               =   GS_MSGCOLL-MSGV1            " Parameter 1
*       MSGV2               =   GS_MSGCOLL-MSGV2            " Parameter 2

*     IMPORTING
*       MESSAGE_TEXT_OUTPUT = GV_MSG.                 " Output message text

    MESSAGE I001 WITH GV_MSG.

 ELSE.
**********************************************************************
*BDC 수행중 ERROR가 발생 했다면 그 위치에서 BDC 수행은 중지되기 때문에
*해당 Error message를 얻어오려면 GT_MESSTAB의 마지막 라인을 읽으면 됨
**********************************************************************
  CLEAR GS_MSGCOLL.
  READ TABLE GT_MSGCOLL INTO GS_MSGCOLL INDEX LINES( GT_MSGCOLL ).

  CLEAR GV_MSG.
*  CALL FUNCTION 'MMESSAGE_TEXT_BUILD'
*      EXPORTING
*       MSGID               =   GS_MSGCOLL-MSGID               " Message ID
*       MSGNR               =   GS_MSGCOLL-MSGNR               " Number of message
*       MSGV1               =   GS_MSGCOLL-MSGV1            " Parameter 1
*       MSGV2               =   GS_MSGCOLL-MSGV2            " Parameter 2
*       MSGV3               =   GS_MSGCOLL-MSGV3            " Parameter 3
*       MSGV4               =   GS_MSGCOLL-MSGV4            " Parameter 4
*     IMPORTING
*       MESSAGE_TEXT_OUTPUT = GV_MSG.                 " Output message text

    _MSG_BUILD GS_MSGCOLL-MSGID GS_MSGCOLL-MSGNR
               GS_MSGCOLL-MSGV1 GS_MSGCOLL-MSGV2 GS_MSGCOLL-MSGV3 GS_MSGCOLL-MSGV4
               GV_MSG.


    MESSAGE I001 WITH GV_MSG DISPLAY LIKE 'E'.


 ENDIF.


ENDFORM.

FORM DYNPRO USING PV_BEGIN PV_FIELD PV_VALUE.

  CASE PV_BEGIN.
    WHEN 'X'. "SCREEN BEGIN
      GS_BDCDATA-DYNBEGIN = PV_BEGIN. "SCREEN START
      GS_BDCDATA-PROGRAM  = PV_FIELD. "PROGRAM ID
      GS_BDCDATA-DYNPRO   = PV_VALUE. "SCREEN NO
    WHEN OTHERS.
       GS_BDCDATA-FNAM = PV_FIELD. "Field Name
       GS_BDCDATA-FVAL = PV_VALUE. "Field Value
  ENDCASE.

      APPEND GS_BDCDATA TO GT_BDCDATA.
      CLEAR GS_BDCDATA.

ENDFORM.
