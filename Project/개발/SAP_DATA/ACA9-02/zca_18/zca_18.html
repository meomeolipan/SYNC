REPORT ZCA_18.

include ZCA_18_TOP.
include ZCA_18_SCR.
include ZCA_18_F01.

INITIALIZATION.

AT SELECTION-SCREEN.

START-OF-SELECTION.

IF PA_TAB IS INITIAL.
  MESSAGE '테이블 이름이 없습니다.' TYPE 'E'.
  ELSE.
    CASE PA_TAB.
      WHEN 'SCARR'.
          ASSIGN GT_SCARR TO <FS_TAB>.
      WHEN 'SPFLI'.
         ASSIGN GT_SPFLI TO <FS_TAB>.
      WHEN 'SFLIGHT'.
         ASSIGN GT_SFLIGHT TO <FS_TAB>.
      WHEN 'SBOOK'.
         ASSIGN GT_SBOOK TO <FS_TAB>.
    ENDCASE.

  IF <FS_TAB> IS ASSIGNED.
    SELECT * FROM (PA_TAB) INTO TABLE <FS_TAB>.

      PERFORM DISPLAY_SALV USING <FS_TAB>.
  ELSE.
    MESSAGE '테이블 이름이 잘못되었습니다' TYPE 'I' DISPLAY LIKE 'E'.

  ENDIF.


ENDIF.
