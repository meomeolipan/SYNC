TYPES: TY_SPFLI TYPE STANDARD TABLE OF SPFLI
                     WITH NON-UNIQUE KEY CARRID CONNID.

DATA OK_CODE TYPE SY-UCOMM.

DATA: GT_SPFLI TYPE TY_SPFLI.

* ALV 를 위한 참조변수 선언
DATA: GO_CONTAINER TYPE REF TO CL_GUI_CUSTOM_CONTAINER,
      GO_ALV_GRID  TYPE REF TO CL_GUI_ALV_GRID.

*--------------------------------------------------------------------*

* LCL_EVENT_HANDLER 클래스를 정의구현한다. Double 클릭을 위한 메소드로
* ON_DOUBLE_CLICK 이라는 이름을 가진 Instance Method를 선언하고 구현
* 구현할 내용은 사용자가 Double Click 시 해당 라인의 행번호와 열이름을
* 팝업으로 출력한다.

CLASS LCL_EVENT_HANDLER DEFINITION.
  PUBLIC SECTION.
    METHODS ON_DOUBLE_CLICK FOR EVENT DOUBLE_CLICK
                                   OF CL_GUI_ALV_GRID
                            IMPORTING ES_ROW_NO E_COLUMN.
  PRIVATE SECTION.
ENDCLASS.

CLASS LCL_EVENT_HANDLER IMPLEMENTATION.
  METHOD ON_DOUBLE_CLICK.
    DATA LV_MSG TYPE STRING.
    DATA LV_ROW_ID TYPE STRING.

    " 형 변환 ( TYPE CASTING )
    LV_ROW_ID = ES_ROW_NO-ROW_ID.

    CONCATENATE '더블 클릭한 행 번호는' LV_ROW_ID
                ', 그리고 열 이름은'    E_COLUMN-FIELDNAME
                '입니다.'
           INTO LV_MSG SEPARATED BY SPACE.

    MESSAGE I001(ZMSG_E00) WITH LV_MSG.
*    MESSAGE LV_MSG TYPE I.
  ENDMETHOD.
ENDCLASS.


DATA GO_HANDLER TYPE REF TO LCL_EVENT_HANDLER.

*--------------------------------------------------------------------*


START-OF-SELECTION.

  SELECT * FROM SPFLI INTO TABLE GT_SPFLI.

  CALL SCREEN '0100'.

MODULE STATUS_0100 OUTPUT.

*&---------------------------------------------------------------------*
*&      Module  ALV_GRID  OUTPUT
*&---------------------------------------------------------------------*
MODULE ALV_GRID OUTPUT.

  IF GO_CONTAINER IS INITIAL.
*** Create object of class CL_GUI_CUSTOM_CAONTAINER to manage data !
    CREATE OBJECT GO_CONTAINER
      EXPORTING
        CONTAINER_NAME              = 'GO_CON'                 " Name of the Screen CustCtrl Name to Link Container To
      EXCEPTIONS
        CNTL_ERROR                  = 1                " CNTL_ERROR
        CNTL_SYSTEM_ERROR           = 2                " CNTL_SYSTEM_ERROR
        CREATE_ERROR                = 3                " CREATE_ERROR
        LIFETIME_ERROR              = 4                " LIFETIME_ERROR
        LIFETIME_DYNPRO_DYNPRO_LINK = 5                " LIFETIME_DYNPRO_DYNPRO_LINK
        OTHERS                      = 6.

    CASE SY-SUBRC.
      WHEN 0.
        " 정상
      WHEN OTHERS.
        " 오류
        " Custom Container 생성 중 오류가 발생했습니다.
        MESSAGE E020(ZMSG_E00).
    ENDCASE.

*** create object of class cl_gui_alv_grid to visualize data !
    CREATE OBJECT GO_ALV_GRID
      EXPORTING
        I_PARENT          = GO_CONTAINER " Parent Container
      EXCEPTIONS
        ERROR_CNTL_CREATE = 1 " Error when creating the control
        ERROR_CNTL_INIT   = 2 " Error While Initializing Control
        ERROR_CNTL_LINK   = 3 " Error While Linking Control
        ERROR_DP_CREATE   = 4 " Error While Creating DataProvider Control
        OTHERS            = 5.

    CASE SY-SUBRC.
      WHEN 0.
        " 정상
      WHEN OTHERS.
        " 오류
        " ALV Grid 생성 중 오류가 발생했습니다.
        MESSAGE E021(ZMSG_E00).
    ENDCASE.

*** create handler instance and register handler method ********

    " 핸들러 인스턴스를 생성
    IF GO_HANDLER IS INITIAL.
      CREATE OBJECT GO_HANDLER.
    ENDIF.

    " 핸들러 메소드를 등록
    " ALV Grid만 대상으로 DOUBLE_CLICK 이벤트가 발생할 때
    " GO_HANDLER의 ON_DOUBLE_CLICK 메소드가 자동으로 실행된다.
    SET HANDLER GO_HANDLER->ON_DOUBLE_CLICK FOR GO_ALV_GRID.


*** Call method to visualize data of internal table ************
    CALL METHOD GO_ALV_GRID->SET_TABLE_FOR_FIRST_DISPLAY
      EXPORTING
        I_STRUCTURE_NAME              = 'SPFLI'                 " Internal Output Table Structure Name
      CHANGING
        IT_OUTTAB                     = GT_SPFLI                 " Output Table
      EXCEPTIONS
        INVALID_PARAMETER_COMBINATION = 1                " Wrong Parameter
        PROGRAM_ERROR                 = 2                " Program Errors
        TOO_MANY_LINES                = 3                " Too many Rows in Ready for Input Grid
        OTHERS                        = 4.

    CASE SY-SUBRC.
      WHEN 0.
        " 정상
*      WHEN 1.
*      WHEN 2.
*      WHEN 3.
      WHEN OTHERS.
        " ALV Grid 에 데이터를 전달하는 중 오류가 발생했습니다.
        MESSAGE E023(ZMSG_E00).

    ENDCASE.


  ENDIF.

ENDMODULE. " ALV_GRID OUTPUT
