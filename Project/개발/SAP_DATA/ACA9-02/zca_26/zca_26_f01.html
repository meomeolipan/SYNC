*&---------------------------------------------------------------------*
FORM MAKE_DISPLAY_BODY .

 DATA : LV_TABIX TYPE SY-TABIX,
        LS_SCOL  TYPE LVC_S_SCOL.

 LOOP AT GT_SCARR INTO GS_SCARR.

  LV_TABIX = SY-TABIX.

  CLEAR: GS_SCARR-COLOR, LS_SCOL.

  CASE GS_SCARR-CURRCODE.
   WHEN 'USD'.
     LS_SCOL-FNAME = 'URL'.
     LS_SCOL-COLOR-COL = 6.
     LS_SCOL-COLOR-INT = 0.
     LS_SCOL-COLOR-INV = 1.
     APPEND LS_SCOL TO GS_SCARR-COLOR.

     CLEAR LS_SCOL.
     LS_SCOL-FNAME = 'CURRCODE'.
     LS_SCOL-COLOR-COL = 6.
     APPEND LS_SCOL TO GS_SCARR-COLOR.

   WHEN 'EUR'.
     " FNAME을 안주면 그냥 라인 전부 색칠함!!
     LS_SCOL-COLOR-COL = 3.
     LS_SCOL-COLOR-INT = 1.
     LS_SCOL-COLOR-INV = 0.
     LS_SCOL-NOKEYCOL = ABAP_ON.
     " ALV에서 Key Column은 파란색 칠해져있는데 그거 무시하는거임
     APPEND LS_SCOL TO GS_SCARR-COLOR.
   WHEN OTHERS.
     LS_SCOL-FNAME = 'CARRNAME'.
     LS_SCOL-COLOR-COL = 7.
     LS_SCOL-COLOR-INT = 1.
     " 스위치 켤거만 값 넣어주면된다 INV 안넣을거면 걍 냅둬라
     APPEND LS_SCOL TO GS_SCARR-COLOR.

  ENDCASE.

  MODIFY GT_SCARR FROM GS_SCARR INDEX LV_TABIX TRANSPORTING COLOR.

 ENDLOOP.


ENDFORM.
