*& <--  p2        text

FORM GET_AIRLINE_MASTER .

  CLEAR GT_TOP.
  SELECT CARRID CARRNAME CURRCODE URL
    INTO CORRESPONDING FIELDS OF TABLE GT_TOP
    FROM SCARR
    WHERE CARRID IN SO_CARR.

    IF GT_TOP IS INITIAL.
      MESSAGE S003 DISPLAY LIKE 'E'.
      STOP.
    ENDIF.


ENDFORM.

*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*&      --> E_COLUMN
*&      --> E_ROW
*&      --> ES_ROW_NO
*&---------------------------------------------------------------------*
FORM HANDLE_DOUBLE_CLICK USING PE_ROW PE_COLUMN.


    CLEAR GS_TOP.
  READ TABLE GT_TOP INTO GS_TOP INDEX PE_ROW.

  CLEAR GT_BOT.
  SELECT *
    INTO CORRESPONDING FIELDS OF TABLE GT_BOT
    FROM SPFLI
   WHERE CARRID = GS_TOP-CARRID.

  IF GT_BOT IS INITIAL.
    MESSAGE S003 DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  PERFORM REFRESH_TABLE.
  LEAVE SCREEN.

ENDFORM.
