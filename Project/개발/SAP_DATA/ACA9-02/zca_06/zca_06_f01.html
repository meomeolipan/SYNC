FORM SELECT_DATA .

  CLEAR GT_TABLE.


  SELECT A~CARRID,
         A~CARRNAME,
         B~CONNID,
         B~COUNTRYFR,
         B~CITYFROM,
         B~COUNTRYTO,
         B~CITYTO,
         C~FLDATE,
         C~PRICE,
         C~CURRENCY,
         C~SEATSMAX,
         C~SEATSOCC
    FROM SCARR AS A
    JOIN SPFLI AS B
    ON A~CARRID EQ B~CARRID
    JOIN SFLIGHT AS C
    ON B~CARRID EQ C~CARRID
    AND B~CONNID EQ C~CONNID

    WHERE C~FLDATE IN @S_FLI
    AND B~COUNTRYFR EQ @A_CON
    AND B~CITYFROM IN @A_CIT
    AND B~COUNTRYTO EQ @D_CON
    AND B~CITYTO  IN @D_CIT
    ORDER BY C~FLDATE ASCENDING
    INTO CORRESPONDING FIELDS OF TABLE @GT_TABLE.










ENDFORM.

FORM MODIFY_DATA .


  LOOP AT GT_TABLE INTO GS_TABLE.

    GS_TABLE-SEATFRE = GS_TABLE-SEATSMAX - GS_TABLE-SEATSOCC.

    IF GS_TABLE-SEATFRE > 200.
      GS_TABLE-LIGHT = ICON_GREEN_LIGHT.

    ELSEIF GS_TABLE-SEATFRE <= 10.

       GS_TABLE-LIGHT = ICON_RED_LIGHT.

    ELSE.

       GS_TABLE-LIGHT = ICON_YELLOW_LIGHT.

    ENDIF.

    MODIFY GT_TABLE FROM GS_TABLE.


  ENDLOOP.

ENDFORM.
