*&---------------------------------------------------------------------*
FORM DELE_ROW .


  DATA : LT_ROW  TYPE LVC_T_ROW,
         LT_ROID TYPE LVC_T_ROID,
         LS_ROW  TYPE LVC_S_ROW,
         LS_ROID TYPE LVC_S_ROID.

*-- 사용자가 선택한 행 정보를 얻는다
  CALL METHOD GO_ALV_GRID->GET_SELECTED_ROWS
    IMPORTING
      ET_INDEX_ROWS = LT_ROW
      ET_ROW_NO     = LT_ROID.

*-- 사용자가 행을 선택하지 않고 Delete row 버튼을 Click 했을경우
  IF LT_ROW IS INITIAL.
    MESSAGE S001 WITH TEXT-E01 DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

*-- Delete row
  SORT LT_ROW BY INDEX DESCENDING.

  LOOP AT LT_ROW INTO LS_ROW.

*-- 삭제전 삭제용 ITAB에 데이터 이관
    CLEAR GS_BODY.
    READ TABLE GT_BODY INTO GS_BODY INDEX LS_ROW-INDEX.

    MOVE-CORRESPONDING GS_BODY TO GS_DELT.
    APPEND GS_DELT TO GT_DELT.
    CLEAR GS_DELT.

*-- 선택행 삭제
    DELETE GT_BODY INDEX LS_ROW-INDEX.

  ENDLOOP.

ENDFORM.
