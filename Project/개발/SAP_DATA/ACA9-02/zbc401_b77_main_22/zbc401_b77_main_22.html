    DATA: LS_PLANETYPE TYPE SAPLANE.

    MV_NAME      = IV_NAME.

    SELECT SINGLE * FROM SAPLANE
                    INTO LS_PLANETYPE
                   WHERE PLANETYPE = IV_PLANETYPE.

    IF SY-SUBRC <> 0.

      MV_TANKCAP = LS_PLANETYPE-TANKCAP.

    ENDIF.
