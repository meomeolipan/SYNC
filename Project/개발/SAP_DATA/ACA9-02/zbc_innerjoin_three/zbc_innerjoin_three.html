Description: Program ZBC_INNERJOIN_THREE
*&---------------------------------------------------------------------*
*& Report ZBC_INNERJOIN_THREE
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
REPORT ZBC_INNERJOIN_THREE.

TYPES: BEGIN OF GTY_JOIN,
         CARRID     TYPE SPFLI-CARRID,
         CONNID     TYPE SPFLI-CONNID,

         FLDATE     TYPE SFLIGHT-FLDATE,

         CITYFROM   TYPE SPFLI-CITYFROM,
         CITYTO     TYPE SPFLI-CITYTO,
         FLTIME     TYPE SPFLI-FLTIME,

         SEATSMAX   TYPE SFLIGHT-SEATSMAX,
         SEATSOCC   TYPE SFLIGHT-SEATSOCC,

         PLANETYPE  TYPE SAPLANE-PLANETYPE,
         OP_SPEED   TYPE SAPLANE-OP_SPEED,
         SPEED_UNIT TYPE SAPLANE-SPEED_UNIT,
       END OF GTY_JOIN.

DATA: GT_JOIN TYPE TABLE OF GTY_JOIN,
      GS_JOIN TYPE          GTY_JOIN.

DATA: GO_ALV_GRID TYPE REF TO CL_SALV_TABLE.

SELECT-OPTIONS: SO_CAR FOR GS_JOIN-CARRID.


*----------------------------------------------------------------------*
START-OF-SELECTION.
* retrieve data
  SELECT
      A~CARRID A~CONNID
      B~FLDATE
      A~CITYFROM A~CITYTO A~FLTIME
      B~SEATSMAX B~SEATSOCC
     C~PLANETYPE C~OP_SPEED C~SPEED_UNIT
    INTO CORRESPONDING FIELDS OF TABLE GT_JOIN
    FROM SPFLI AS A JOIN SFLIGHT AS B
    ON A~CARRID = B~CARRID AND
       A~CONNID = B~CONNID
   JOIN SAPLANE AS C
    ON B~PLANETYPE = C~PLANETYPE
    WHERE A~CARRID IN SO_CAR.

* create ALV
  CL_SALV_TABLE=>FACTORY(
    IMPORTING
      R_SALV_TABLE   = GO_ALV_GRID
    CHANGING
      T_TABLE        = GT_JOIN ).

* display data
  GO_ALV_GRID->DISPLAY( ).
