FORM get_data .
  SELECT a~carrid a~carrname b~connid b~countryfr b~countryto
    c~fldate c~customid c~passname c~bookid c~order_date c~smoker c~cancelled
    INTO CORRESPONDING FIELDS OF TABLE gt_flight
    FROM scarr AS a
    INNER JOIN spfli AS b
    ON a~carrid = b~carrid
    INNER JOIN sbook AS c
    ON b~carrid = c~carrid
    AND b~connid = c~connid
    WHERE a~carrid = pa_car
    AND b~connid IN so_con
    AND c~fldate IN so_fld
    AND c~customID IN so_cus
    AND c~order_date IN so_ord.

ENDFORM.
*&---------------------------------------------------------------------*
*& Form make_data
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM make_data .

  LOOP AT gt_flight INTO gs_flight.
    "  Status icon 설정
    " 이미 출발한 비행기 체크표시
    " 필드명 일정완료여부

    "    MOVE-CORRESPONDING as_flight TO gs_flight.

    IF gs_flight-fldate < sy-datum.
      gs_flight-status = icon_space.
    ELSE.
      gs_flight-status = icon_okay.
    ENDIF.

    " Light에 취소된 예약은 빨간 아이콘, 그 외에는 초록 아이콘
    " 필드명 예약상태
    IF gs_flight-cancelled = 'X'.
      gs_flight-row_color = 'C' && col_negative && '10'.
      gs_flight-light = '1'.
    ELSE.
      gs_flight-light = '3'.
    ENDIF.

    " SMOKER는 셀 색깔 추가 +  체크박스로 체크
    " 취소된 예약에 관해서는 해당 라인을 빨간색 컬러

    IF gs_flight-smoker = 'X'.
      gs_colfields-fname = 'SMOKER'.
      gs_colfields-color-col = col_group.
      gs_colfields-color-int  = 1.
      gs_colfields-color-inv  = 0.
      gs_colfields-nokeycol = 'X'.
      APPEND gs_colfields TO gs_flight-cell_color.
    ENDIF.

    MODIFY gt_flight FROM gs_flight TRANSPORTING status cell_color light row_color.
  ENDLOOP.
ENDFORM.

*& text
*&---------------------------------------------------------------------*
*&      --> E_UCOMM
*&---------------------------------------------------------------------*
FORM handle_user_command USING p_e_ucomm.
  CASE p_e_ucomm.
    WHEN 'REFRESH'.
      REFRESH gt_filter.

    WHEN 'SMOKER'.
      CLEAR gs_filter.
      gs_filter-fieldname = 'SMOKER'.
      gs_filter-sign = 'I'.
      gs_filter-option = 'EQ'.
      gs_filter-low = 'X'.
      APPEND gs_filter TO gt_filter.

    WHEN 'CANCELLED'.
      CLEAR gs_filter.
      gs_filter-fieldname = 'CANCELLED'.
      gs_filter-sign = 'I'.
      gs_filter-option = 'EQ'.
      gs_filter-low = 'X'.
      APPEND gs_filter TO gt_filter.

  ENDCASE.

  PERFORM display_alv.

ENDFORM.
