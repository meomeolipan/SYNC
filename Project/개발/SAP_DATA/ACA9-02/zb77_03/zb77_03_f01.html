*&---------------------------------------------------------------------*
*& Form HANDLE_DOUBLE_CLICK
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*&      --> E_ROW
*&      --> E_COLUMN
*&---------------------------------------------------------------------*
FORM HANDLE_DOUBLE_CLICK USING PE_ROW
                                   PE_COLUMN.

  CLEAR GS_TOP.
  READ TABLE GT_TOP INTO GS_TOP INDEX PE_ROW.

  CLEAR GT_BOT.
  SELECT * INTO CORRESPONDING FIELDS OF TABLE GT_BOT
  FROM BSEG
  WHERE BUKRS = GS_TOP-BUKRS
    AND BELNR = GS_TOP-BELNR
    AND GJAHR = GS_TOP-GJAHR.

**********************************************************************
   CLEAR GT_CSKT.
  SELECT * INTO CORRESPONDING FIELDS OF TABLE GT_CSKT
    FROM CSKT
    WHERE SPRAS EQ SY-LANGU.

* **********************************************************************
    DATA LV_TABIX TYPE SY-TABIX.

  SORT GT_CSKT BY KOSTL ASCENDING.

  LOOP AT GT_BOT INTO GS_BOT.

    LV_TABIX = SY-TABIX.

    CLEAR GS_CSKT.
    READ TABLE GT_CSKT INTO GS_CSKT WITH KEY KOSTL = GS_BOT-KOSTL
                                                   BINARY SEARCH.

    IF SY-SUBRC EQ 0.
        GS_BOT-KTEXT = GS_CSKT-KTEXT.

    ENDIF.

    MODIFY GT_BOT FROM GS_BOT.

  ENDLOOP.




 IF GT_BOT IS INITIAL.
   MESSAGE S003 DISPLAY LIKE 'E'.
   EXIT.
 ENDIF.

 PERFORM REFRESH_TABLE.
 LEAVE SCREEN.

ENDFORM.
*&---------------------------------------------------------------------*
*& Form MAKE_DISPLAY_TOP
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM MAKE_DISPLAY_TOP .


 DATA : LV_TABIX TYPE SY-TABIX,
        LS_SCOL  TYPE LVC_S_SCOL.


  LOOP AT GT_TOP INTO GS_TOP.

    LV_TABIX = SY-TABIX.

  CLEAR: GS_TOP-CELLCOL, LS_SCOL.

  CASE GS_TOP-WAERS.
    WHEN 'KRW'.

     CLEAR LS_SCOL.
     LS_SCOL-FNAME = 'WAERS'.
     LS_SCOL-COLOR-COL = 1.
     APPEND LS_SCOL TO GS_TOP-CELLCOL.

    WHEN 'EUR'.
     CLEAR LS_SCOL.
     LS_SCOL-FNAME = 'WAERS'.
     LS_SCOL-COLOR-COL = 6.
     APPEND LS_SCOL TO GS_TOP-CELLCOL.

  ENDCASE.

   MODIFY GT_TOP FROM GS_TOP INDEX LV_TABIX TRANSPORTING CELLCOL.


  ENDLOOP.

ENDFORM.
