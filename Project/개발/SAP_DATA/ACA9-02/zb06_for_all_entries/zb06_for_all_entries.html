SELECT-OPTIONS SO_STUID FOR ZTCLASS_B06-STUDENTID NO INTERVALS NO-EXTENSION. " 학생ID 입력.

" 검색 조건의 역할.
SELECT SUBJECTNO
  INTO CORRESPONDING FIELDS OF TABLE GT_CONDITION
  FROM ZTCLASS_B06
   WHERE STUDENTID  IN  SO_STUID
     AND APP_GRADE = 1
     AND APP_SEMESTER = 2 .

SORT GT_CONDITION BY SUBJECTNO.
DELETE ADJACENT DUPLICATES FROM GT_CONDITION COMPARING SUBJECTNO.


IF GT_CONDITION IS NOT INITIAL.

* 실제로 검색되어야 할 table
  SELECT SUBJECT_NO SUBJECT CLOSURE INTO TABLE GT_SOURCE
              FROM ZTSUBJECT_B06 FOR ALL ENTRIES IN GT_CONDITION
             WHERE SUBJECT_NO = GT_CONDITION-SUBJECTNO.

ENDIF.
