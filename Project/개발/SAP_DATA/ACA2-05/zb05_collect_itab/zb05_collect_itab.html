Description: collect itab
*&---------------------------------------------------------------------*
*& Report ZB00_COLLECT_ITAB
*&---------------------------------------------------------------------*
*& COLLECT 구문을 활용할 때는
*&     sorted (index, key access) 나
*&     hashed (key access) 를 사용하는 것이 performance 유리하다.
*&     UNIQUE KEY 설정할 것
*&---------------------------------------------------------------------*
REPORT zb05_collect_itab.

DATA : go_alv_grid TYPE REF TO cl_salv_table.

*- 데이터를 조회하기 위한 선언
TYPES : BEGIN OF gty_s_flight,
          carrid   TYPE sflight-carrid,
          connid   TYPE sflight-connid,
          fldate   TYPE sflight-fldate,
          seatsocc TYPE sflight-seatsocc,    " 합계에 계산되어야 하는 값
        END OF gty_s_flight.

DATA : gt_flight TYPE TABLE OF gty_s_flight,
       gs_flight TYPE gty_s_flight.

*- collect 를 위한 선언 (기준키를 순서대로 나열)
TYPES : BEGIN OF gty_s_sum,
          carrid      TYPE sflight-carrid,  "기준키
          connid      TYPE sflight-connid,  "기준키

          seatsocc    TYPE sflight-seatsocc, " 합계에 계산되어야 하는 값
          n_o_flights TYPE i,                " 갯수에 계산되어야 하는 값
        END OF gty_s_sum.

DATA : gt_sum TYPE TABLE OF gty_s_sum,
       gs_sum TYPE gty_s_sum.


START-OF-SELECTION.

  SELECT carrid connid fldate seatsocc
     INTO TABLE gt_flight
           FROM sflight.

  PERFORM collect_loop USING gt_flight CHANGING gt_sum.

  " 결과 출력
  cl_salv_table=>factory(
    IMPORTING
      r_salv_table   = go_alv_grid
    CHANGING
      t_table        = gt_sum ).

  go_alv_grid->display( ).




*&---------------------------------------------------------------------*
*& Form collect_loop
*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*&      --> GT_FLIGHT
*&      <-- GT_SUM
*&---------------------------------------------------------------------*
FORM collect_loop USING pt_flight LIKE gt_flight
                   CHANGING ct_sum    LIKE gt_sum.

  DATA : ls_flight LIKE LINE OF pt_flight,  "WA
         ls_sum    LIKE LINE OF ct_sum,     "collect wa
         lt_sorted LIKE SORTED TABLE OF ls_sum
                   WITH UNIQUE DEFAULT KEY,  "collect itab
         lt_hashed LIKE HASHED TABLE OF ls_sum
                   WITH UNIQUE DEFAULT KEY.   "<- performance에 유리

*- 1) Hashed
  LOOP AT pt_flight INTO ls_flight.
    MOVE-CORRESPONDING ls_flight TO ls_sum.  "데이터 밀어넣고
    ls_sum-n_o_flights = 1.                  " 카운팅 1씩 늘리고
    COLLECT ls_sum INTO lt_hashed.           " sum 하면서 append하는 개념의 COLLECT 적용
  ENDLOOP.

  SORT ct_sum.
  ct_sum = lt_hashed.  " 결과 리턴

**- 2) Sorted
*  LOOP AT pt_flight INTO ls_flight.
*    MOVE-CORRESPONDING ls_flight TO ls_sum.  "데이터 밀어넣고
*    ls_sum-n_o_flights = 1.                  " 키 기준으로 카운팅 1씩 늘리고
*    COLLECT ls_sum INTO lt_sorted.           " sum 하면서 append하는 개념의 COLLECT 적용
*  ENDLOOP.
*
*  ct_sum = lt_sorted.  " 결과 리턴

ENDFORM.
