FUNCTION zfmb05_get_bezei.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" REFERENCE(IV_AUART) TYPE AUART
*" EXPORTING
*" REFERENCE(EV_BEZEI) TYPE TVAKT-BEZEI
*" EXCEPTIONS
*" NO_DATA
*"----------------------------------------------------------------------

* Global data declarations

  SELECT SINGLE bezei INTO ev_bezei
    FROM tvakt
    WHERE auart = iv_auart
    AND spras EQ sy-langu.

  IF sy-subrc <> 0.
    RAISE no_data.
  ENDIF.

ENDFUNCTION.
