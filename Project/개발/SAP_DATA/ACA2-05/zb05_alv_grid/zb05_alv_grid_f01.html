FORM get_data .

  CASE 'X'.
    WHEN pa_r1.
      gv_dynamic = ''.
    WHEN pa_r2.
      gv_dynamic = 'Y'.
    WHEN pa_r3.
      gv_dynamic = 'C'.
    WHEN pa_r4.
      gv_dynamic = 'F'.
    WHEN OTHERS.
  ENDCASE.

  IF gv_dynamic IS INITIAL.

    SELECT * FROM sbook AS a
        LEFT OUTER JOIN scarr AS b ON a~carrid EQ b~carrid
        INTO CORRESPONDING FIELDS OF TABLE gt_sbook
        WHERE a~carrid IN s_car
        AND   a~connid IN s_con
        AND   a~fldate IN s_fld
        AND   a~cancelled = pa_c1.

  ELSE.

    SELECT * FROM sbook AS a
        LEFT OUTER JOIN scarr AS b ON a~carrid EQ b~carrid
        INTO CORRESPONDING FIELDS OF TABLE gt_sbook
        WHERE a~carrid IN s_car
        AND   a~connid IN s_con
        AND   a~fldate IN s_fld
        AND   a~cancelled = pa_c1
        AND   ( a~class =  gv_dynamic ).

  ENDIF.

ENDFORM.

*&---------------------------------------------------------------------*
*& text
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM make_data .

  LOOP AT gt_sbook INTO gs_sbook.

    IF gs_sbook-custtype = 'B'.
      gs_sbook-icon = 3.
    ELSEIF gs_sbook-custtype = 'P'.
      gs_sbook-icon = 2.
    ENDIF.

    IF gs_sbook-smoker = 'X'.
      gs_sbook-row_color = 'C' && col_negative &&    '01'.
    ENDIF.

    IF gs_sbook-invoice = 'X'.
      gs_field_color-fname     = 'INVOICE'.
      gs_field_color-color-col = col_positive.
      gs_field_color-color-int = 1.
      gs_field_color-color-inv = 1.
      gs_field_color-nokeycol = 'X'.
      APPEND gs_field_color TO gs_sbook-it_field_colors.

    ENDIF.

    IF gs_sbook-cancelled = 'X'.
      gs_sbook-change_possible = icon_red_light.
    ELSE.
      gs_sbook-change_possible = icon_green_light.
    ENDIF.

    MODIFY gt_sbook FROM gs_sbook
  TRANSPORTING icon row_color it_field_colors change_possible.

    CLEAR gs_sbook.

  ENDLOOP.

ENDFORM.
