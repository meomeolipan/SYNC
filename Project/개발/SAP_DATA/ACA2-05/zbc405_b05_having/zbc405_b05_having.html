Description: Program ZBC405_B05_HAVING
*&---------------------------------------------------------------------*
*& Report ZBC405_B05_HAVING
*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
REPORT zbc405_b05_having.

TYPES: BEGIN OF gty_sflight,
         carrid       TYPE sflight-carrid,
         seatsmax     TYPE sflight-seatsmax,
         seatsocc     TYPE sflight-seatsocc,
         seatsmax_avg TYPE sflight-seatsmax,
         seatsocc_avg TYPE sflight-seatsocc,
         num_conn     TYPE i,
         num_sflight  TYPE i,
       END OF gty_sflight.

DATA: gt_sflight TYPE TABLE OF gty_sflight,
      gs_sflight TYPE gty_sflight.

DATA: go_alv_grid TYPE REF TO cl_salv_table.

SELECT-OPTIONS: so_car FOR gs_sflight-carrid,
                so_maxav FOR  gs_sflight-seatsmax_avg,
                so_occav  FOR gs_sflight-seatsocc_avg.

START-OF-SELECTION.

  SELECT  carrid
          COUNT( DISTINCT connid ) AS num_conn
          SUM( seatsmax ) AS seatsmax
          SUM( seatsocc ) AS seatsocc
          AVG( seatsmax ) AS seatsmax_avg
          AVG( seatsocc ) AS seatsocc_avg
          COUNT( * ) AS num_sflight
    INTO CORRESPONDING FIELDS OF TABLE gt_sflight
    FROM sflight
    WHERE carrid IN so_car
    GROUP BY carrid
    HAVING AVG( seatsmax ) IN so_maxav
       AND AVG( seatsocc ) IN so_occav
    ORDER BY seatsmax_avg DESCENDING
             seatsocc_avg DESCENDING.

*    TRY.
  CALL METHOD cl_salv_table=>factory
*      EXPORTING
*        list_display   = IF_SALV_C_BOOL_SAP=>FALSE
*        r_container    =
*        container_name =
    IMPORTING
      r_salv_table = go_alv_grid
    CHANGING
      t_table      = gt_sflight.
  .
*      CATCH cx_salv_msg.
*    ENDTRY.

  go_alv_grid->display( ).
