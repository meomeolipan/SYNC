REPORT ZBC405_B05_GROUP_BY2.

tables sflight.

types : begin of gty_sflight,
    carrid type sflight-carrid,
    seatsmax type sflight-seatsmax,
    seatsocc type sflight-seatsocc,
    seatsmax_avg type sflight-seatsmax,
    seatsocc_avg type sflight-seatsocc,
    num_connid type i,
    num_flights type i,
  end of gty_sflight.

  data : gt_sflight type table of gty_sflight,
         gs_sflight type gty_sflight.

  DATA : go_alv_grid TYPE REF TO cl_salv_table.

  select-OPTIONS : so_car for sflight-carrid.
  select-OPTIONS : so_avgmx for sflight-seatsmax.
  select-OPTIONS : so_avgoc for sflight-seatsocc.

start-of-selection.

select carrid
  sum( seatsmax ) as seatsmax
  sum( seatsocc ) as seatsocc
  avg( seatsmax ) as seatsmax_avg
  avg( seatsocc ) as seatsocc_avg
  count( distinct connid ) as num_connid
  count( * ) as num_flights
  from sflight
  into CORRESPONDING FIELDS OF table gt_sflight
  where carrid in so_car
  group by carrid
  having avg( seatsmax ) in so_avgmx and avg( seatsocc ) in so_avgoc
  order by seatsmax_avg DESCENDING seatsocc_avg DESCENDING.

    CHANGING
      t_table      = gt_sflight.
