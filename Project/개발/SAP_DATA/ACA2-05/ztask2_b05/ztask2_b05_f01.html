FORM modify_data .

  REFRESH gt_display.

  LOOP AT gt_data INTO gs_data.

    CLEAR gs_display.
    MOVE-CORRESPONDING gs_data TO gs_display.

    IF gs_display-fldate LE sy-datum.
      gs_display-status = icon_checked.
    ENDIF.


    IF gs_display-cancelled EQ abap_on.
      gs_display-light = '1'. "빨간색
    ELSE.
      gs_display-light = '3'. "초록색
    ENDIF.

    DATA: ls_cell_color LIKE LINE OF gs_display-cell_color.

    IF gs_display-smoker EQ abap_on.
      CLEAR ls_cell_color.
      ls_cell_color-fname = 'SMOKER'.
      ls_cell_color-color-col = 7.
      ls_cell_color-color-int = 1.
      ls_cell_color-color-inv = 0.
      INSERT ls_cell_color INTO TABLE gs_display-cell_color.
    ENDIF.

    IF gs_display-cancelled EQ abap_on.
      CLEAR ls_cell_color.
*      LS_CELL_COLOR-FNAME = 'CANCELLED'.
      ls_cell_color-color-col = 6.
      ls_cell_color-color-int = 1.
      ls_cell_color-color-inv = 0.
      ls_cell_color-nokeycol = abap_on. "ALV에서 Key Column은 파란색 칠해져있는데 그거 무시하는거임

      INSERT ls_cell_color INTO TABLE gs_display-cell_color.

    ENDIF.

    APPEND gs_display TO gt_display.


  ENDLOOP.

ENDFORM.

FORM select_data .

  REFRESH gt_data.

  SELECT a~carrid a~carrname
         b~connid b~countryfr b~countryto
         c~fldate c~customid c~passname c~bookid c~order_date c~smoker c~cancelled
    INTO CORRESPONDING FIELDS OF TABLE gt_data
           FROM scarr     AS a
           JOIN spfli     AS b ON b~carrid EQ a~carrid
           JOIN sbook     AS c ON c~carrid EQ a~carrid
                              AND c~connid EQ b~connid

            WHERE a~carrid      EQ pa_car
              AND b~connid      IN so_con
              AND c~fldate      IN so_fld
              AND c~customid    IN so_cid
              AND c~order_date  IN so_ord
              AND c~passname    NE space.

ENDFORM.

FORM get_fieldcat_0100 .

  DATA lt_fcat TYPE kkblo_t_fieldcat.

    EXPORTING
      i_callback_program     = sy-repid        " 현재 프로그램
      i_tabname              = 'GS_DISPLAY'    " FIELDCAT이 필요한 Structure 변수이름

*& <--  p2        text
*&---------------------------------------------------------------------*
FORM make_fieldcat_0100 .

  LOOP AT gt_fcat INTO gs_fcat.

    CASE gs_fcat-fieldname.
      WHEN 'LIGHT'.
        gs_fcat-coltext = '상태'.
      WHEN 'STATUS'.
        gs_fcat-coltext = '일정완료여부'.
        gs_fcat-just    = 'C'. " L: LEFT, C:CENTER, R:RIGHT
        gs_fcat-col_pos = 3.
      WHEN 'CARRID'.
        " 항공사 ID 는 키필드로 취급되지 않고, 배경색이 진한 초록색이 된다.
*        GS_FCAT-EMPHASIZE = 'C500'.
*        GS_FCAT-KEY       = ABAP_OFF.
      WHEN 'SMOKER'.
        gs_fcat-checkbox = abap_on.
      WHEN 'CANCELLED'.
*        GS_FCAT-CHECKBOX = ABAP_ON.
        gs_fcat-no_out = abap_on.
      WHEN 'BOOKID'.
        gs_fcat-hotspot = abap_on.
    ENDCASE.

    MODIFY gt_fcat FROM gs_fcat.

  ENDLOOP.

ENDFORM.
