FUNCTION zb05_flight.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" REFERENCE(IV_CARRID) TYPE S_CARR_ID
*" EXPORTING
*" REFERENCE(ES_AIRLINE) TYPE ZSSCARR_B05
*" TABLES
*" ET_SPFLI TYPE ZTSPFLI_B05
*" EXCEPTIONS
*" NO_DATA
*"----------------------------------------------------------------------

* Global data declarations

  SELECT SINGLE *
        FROM scarr
        INTO CORRESPONDING FIELDS OF es_airline
        WHERE carrid = iv_carrid.

  IF es_airline IS INITIAL.
    CLEAR es_airline.
    RAISE no_data.
  ELSE.
    SELECT *
      FROM spfli
      INTO CORRESPONDING FIELDS OF TABLE et_spfli
      WHERE carrid = iv_carrid.
  ENDIF.
ENDFUNCTION.
