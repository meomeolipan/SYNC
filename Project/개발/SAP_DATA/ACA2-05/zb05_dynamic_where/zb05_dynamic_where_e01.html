  BREAK-POINT.
* Checking the output parameters
  CASE gc_mark.
    WHEN pa_all.
*     Radiobutton ALL is marked
      gv_dynamic = ''.

    WHEN pa_nat.
*     Radiobutton NATIONAL is marked
      gv_dynamic = 'countryto = dv_flights~countryfr'.

    WHEN pa_int.
*     Radiobutton INTERNATIONAL is marked

  ENDCASE.

  SELECT * FROM dv_flights INTO CORRESPONDING FIELDS OF TABLE gt_flight
  WHERE carrid IN so_car
    AND connid IN so_con
    AND fldate IN so_fdt
