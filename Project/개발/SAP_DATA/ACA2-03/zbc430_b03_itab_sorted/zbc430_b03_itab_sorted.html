REPORT ZBC430_B03_ITAB_SORTED.

* dictionary에 정의된 table type을 활용하여 internal table 변수 선언
DATA : GT_SFLIGHT TYPE ZIT_SFLIGHTB03. "Table type - 실제 database table 아님!
DATA : GS_SFLIGHT TYPE SFLIGHT. "Work Area (= Line Structure) - table과 동일한 structure

* Array Fetch
* Structure로 DB_TABLE 한 줄씩 접근
WRITE / 'Printout in tableorder of Database:'.

SELECT * FROM SFLIGHT INTO GS_SFLIGHT
        WHERE CARRID = 'JL'.
  WRITE : / SY-DBCNT, "Database Table Rows
            GS_SFLIGHT-CARRID,
            GS_SFLIGHT-CONNID,
            GS_SFLIGHT-FLDATE,
            GS_SFLIGHT-PRICE,
            GS_SFLIGHT-CURRENCY,
            GS_SFLIGHT-PLANETYPE.
ENDSELECT.

ULINE.

** 주로 사용 - Database tabe 한 번에 접근하여 속도가 빠름
* Internal Table로 DB_TABLE 여러 줄 접근
SELECT * FROM SFLIGHT INTO TABLE GT_SFLIGHT
         WHERE CARRID = 'JL'.
WRITE / 'Printout in tableorder of sorted ITAB: '.

LOOP AT GT_SFLIGHT INTO GS_SFLIGHT.
  WRITE : / SY-TABIX, "Row Index of Internal Table
            GS_SFLIGHT-CARRID,
            GS_SFLIGHT-CONNID,
            GS_SFLIGHT-FLDATE,
            GS_SFLIGHT-PRICE,
            GS_SFLIGHT-CURRENCY,
            GS_SFLIGHT-PLANETYPE.
ENDLOOP.
