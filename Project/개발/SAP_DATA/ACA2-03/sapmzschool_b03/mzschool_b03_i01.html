MODULE get_studentid INPUT.
  SELECT SINGLE *
    FROM ztstudent_b03
    INTO gs_student
    WHERE stu_id = ztstudent_b03-stu_id.
  CHECK sy-subrc <> 0.
  CLEAR gs_student.
ENDMODULE.

MODULE user_command_0100 INPUT.
  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'FC1' OR 'FC2' OR 'FC3'.
      my_tabstrip-activetab = ok_code.
  ENDCASE.
ENDMODULE.

*&      Module  USER_COMMAND_0110  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0110 INPUT.
  CASE ok_code.
    WHEN 'PUSH'.
      CLEAR gs_class.

      SELECT SINGLE * FROM ztclass_b03
        INTO CORRESPONDING FIELDS OF gs_class
        WHERE stu_id =  ztstudent_b03-stu_id
        AND sub_num = ztsubject_b03-sub_num
        AND sub_grade = ztsubject_b03-sub_grade
        AND apply_semester = ztsubject_b03-sub_semester.


      IF gs_class IS NOT INITIAL.
        MESSAGE '이미 신청한 과목입니다' TYPE 'I'.
      ELSE.
        ztclass_b03-stu_id = ztstudent_b03-stu_id.
        ztclass_b03-sub_num = ztsubject_b03-sub_num.
        ztclass_b03-sub_grade = ztsubject_b03-sub_grade.
        ztclass_b03-apply_semester = ztsubject_b03-sub_SEMESTER.
        ztclass_b03-apply_date = '240220'.
        MODIFY ztclass_b03.
        MESSAGE '신청되었습니다' TYPE 'I'.
      ENDIF.
  ENDCASE.
ENDMODULE.
