FORM get_data .
  "상단 출력 데이터
  SELECT * INTO CORRESPONDING FIELDS OF TABLE gt_flight
    FROM sflight
    WHERE carrid IN so_carid AND
          connid IN so_conid AND
          fldate IN so_fldt.

  LOOP AT gt_flight INTO gs_flight.
    SELECT SINGLE carrname INTO gs_flight-carrname
      FROM scarr
      WHERE carrid = gs_flight-carrid.
    MODIFY gt_flight FROM gs_flight.
  ENDLOOP.
ENDFORM.

FORM make_data .
  LOOP AT gt_book INTO gs_book.
    IF gs_book-class = 'C'.
      gs_book-flight_class = 'Business Class'.
    ELSEIF gs_book-class = 'Y'.
      gs_book-flight_class = 'Economy Class'.
    ELSEIF gs_book-class = 'F'.
      gs_book-flight_class = 'First Class'.
    ENDIF.

    " LIGHT - 예약 여부
    IF gs_book-cancelled = 'X'. "예약 취소
      gs_book-line_color = 'C610'. "라인 색상 빨강
    ENDIF.

    " SMOKER - 흡연 여부
    IF gs_book-smoker = 'X'.
      gs_fieldcol-fname = 'CUSTOMID'.
      gs_fieldcol-color-col = col_negative.
      gs_fieldcol-color-inv = 1. "반전
      gs_fieldcol-color-int = 1. "강조
      APPEND gs_fieldcol TO gs_book-cell_color.
    ENDIF.

    MODIFY gt_book FROM gs_book
         TRANSPORTING flight_class line_color cell_color.
  ENDLOOP.
ENDFORM.

FORM create_grid .
  "container 생성
  IF go_custom IS INITIAL.
*    "메인 컨테이너 선언
*    CREATE OBJECT go_docking
*      EXPORTING
*        repid     = sy-repid
*        dynnr     = sy-dynnr
*        "SIDE                        = DOCK_AT_LEFT
*        extension = 1550.
*
*    "화면 분할 컨테이너 선언
*    CREATE OBJECT go_split
*      EXPORTING
*        parent  = go_docking
*        rows    = 2
*        columns = 1.

     CREATE OBJECT go_custom   "constructor method 가 호출 (IMPORT, EXCEPTIONS)
      EXPORTING
        container_name = 'MY_CONTROL'  "100 스크린에 생성한 이름
      EXCEPTIONS
        OTHERS         = 1.

    CREATE OBJECT go_split
       EXPORTING
            parent  = go_custom
            rows    = 2
            columns = 1.

    "각 분할 영역 컨테이너에 매핑
    CALL METHOD go_split->get_container
      EXPORTING
        row       = 1
        column    = 1
      RECEIVING
        container = go_cont1.
    CALL METHOD go_split->get_container
      EXPORTING
        row       = 2
        column    = 1
      RECEIVING
        container = go_cont2.

    "grid 생성
    CREATE OBJECT go_grid1
      EXPORTING
        i_parent = go_cont1
        i_name   = '항공편 정보'.

    CREATE OBJECT go_grid2
      EXPORTING
        i_parent = go_cont2
        i_name   = '항공편별 예약 정보 '.
  ENDIF.
ENDFORM.

FORM double_click USING e_row TYPE lvc_s_row
                          e_column TYPE lvc_s_col
                          es_row_no TYPE lvc_s_roid.
  DATA: lv_bookings   TYPE i.

  "사용자가 선택한 row를 index 조건으로 데이터 조회
  READ TABLE gt_flight INTO gs_flight INDEX es_row_no-row_id.

  IF sy-subrc = 0.
    "하단 출력 데이터
    SELECT  * INTO CORRESPONDING FIELDS OF TABLE gt_book
     FROM sbook
     WHERE carrid EQ gs_flight-carrid AND
           connid EQ gs_flight-connid AND
           fldate EQ gs_flight-fldate.
    PERFORM make_data.
    PERFORM  set_field_catalog2.
    PERFORM display_grid_bot.
  ELSE.
    MESSAGE s001 DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

ENDFORM.
