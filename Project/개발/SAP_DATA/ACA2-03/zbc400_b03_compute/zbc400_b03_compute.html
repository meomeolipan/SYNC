*&---------------------------------------------------------------------*

*&---------------------------------------------------------------------*

*&---------------------------------------------------------------------*
REPORT ZBC400_B03_COMPUTE.

* 변수 선언
PARAMETERS: PA_INT1 TYPE I,
            PA_INT2  TYPE I,
            PA_OP(1) TYPE C.
DATA: GV_RESULT TYPE P LENGTH 16 DECIMALS 2.

* 사칙연산 기호 입력일 경우
IF PA_OP = '+' OR PA_OP = '-' OR PA_OP = '/' OR PA_OP = '*'.
  " 0으로 나누는 경우
  IF PA_OP = '/' AND PA_INT2 = 0.
    WRITE TEXT-E01.
  ELSE.
    " 사칙연산 결과를 GV_RESULT에 저장
    CASE PA_OP.
      WHEN '+'.
        GV_RESULT = PA_INT1 + PA_INT2.
      WHEN '-'.
        GV_RESULT = PA_INT1 - PA_INT2.
      WHEN '*'.
        GV_RESULT = PA_INT1 * PA_INT2.
      WHEN '/'.
        GV_RESULT = PA_INT1 / PA_INT2.
    ENDCASE.
    " 결과 출력
    WRITE GV_RESULT.
  ENDIF.
* 사칙연산 기호 입력이 아닐 경우
ELSE.
  WRITE TEXT-E02.
ENDIF.

*Text elements
*----------------------------------------------------------
* E01 DIVISION 0 IS ERROR!
* E02 Invalid Arithmetic Operator!


*Selection texts
*----------------------------------------------------------
* PA_INT1         값 1
* PA_INT2         값 2
* PA_OP         기호
