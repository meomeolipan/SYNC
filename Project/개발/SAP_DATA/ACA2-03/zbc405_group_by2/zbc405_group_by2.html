*&---------------------------------------------------------------------*
*&
*&---------------------------------------------------------------------*
REPORT zbc405_group_by2.
** 선언부
DATA: go_alv_grid TYPE REF TO cl_salv_table.

"출력필드 타입 선언
TYPES: BEGIN OF gty_sflight,
         carrid       TYPE sflight-carrid,
         seatsmax     TYPE sflight-seatsmax,
         seatsocc     TYPE sflight-seatsocc,
         seatsmax_avg TYPE sflight-seatsmax,
         seatsocc_avg TYPE sflight-seatsocc,
         num_connid   TYPE i,
         num_flights  TYPE i,
       END OF gty_sflight.

DATA: gt_sflights TYPE TABLE OF gty_sflight,
      gs_sflight  TYPE gty_sflight.

** 입력부
SELECT-OPTIONS: so_car FOR gs_sflight-carrid,
                so_avgmx FOR gs_sflight-seatsmax,
                so_avgoc FOR gs_sflight-seatsocc.

** 구현부
START-OF-SELECTION.
  SELECT carrid "항공사 ID
    SUM( seatsmax ) AS seatsmax "이코노미 좌석수 합계
    SUM( seatsocc ) AS seatsocc "이코노미 예약 좌석수 합계
    AVG( seatsmax ) AS seatsmax_avg "이코노미 좌석수 평균
    AVG( seatsocc ) AS seatsocc_avg "이코노미 예약 좌석수 평균
    COUNT( DISTINCT connid ) AS num_connid "연결항공편 개수-항공사별 항공기 중복 제거
    COUNT(*) AS num_flights "전체 항공편 개수

    INTO CORRESPONDING FIELDS OF TABLE gt_sflights
    FROM sflight
    WHERE carrid IN so_car "사용자 입력 항공사 조건

    GROUP BY carrid "항공사별 그룹화
    HAVING AVG( seatsmax ) IN so_avgmx "좌석/예악좌석 개수 조건을 갖는 그룹
       AND AVG( seatsocc ) IN so_avgoc

    ORDER BY seatsmax_avg DESCENDING "정렬
              seatsocc_avg DESCENDING.

  "출력
  cl_salv_table=>factory(
  IMPORTING
    r_salv_table =  go_alv_grid
    CHANGING
      t_table = gt_sflights ).
  go_alv_grid->display( ).
