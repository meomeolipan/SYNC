FORM get_data .
  CASE 'X'.
    WHEN p_all. "모든 좌석 조회
      SELECT *
         INTO CORRESPONDING FIELDS OF TABLE gt_sbooks
         FROM sbook
         WHERE carrid IN so_carid
         AND connid IN so_conid
         AND fldate IN so_fdate

         AND cancelled EQ p_can.
    WHEN p_fst.
      SELECT *
         INTO CORRESPONDING FIELDS OF TABLE gt_sbooks
         FROM sbook
         WHERE carrid IN so_carid
         AND connid IN so_conid
         AND fldate IN so_fdate

         AND cancelled EQ p_can

        AND CLASS EQ 'F'. "FIRST 좌석 조회
    WHEN p_bus.
      SELECT *
         INTO CORRESPONDING FIELDS OF TABLE gt_sbooks
         FROM sbook
         WHERE carrid IN so_carid
         AND connid IN so_conid
         AND fldate IN so_fdate

         AND cancelled EQ p_can

        AND CLASS EQ 'C'. "BUSNIESS 좌석 조회
    WHEN OTHERS.
      SELECT *
         INTO CORRESPONDING FIELDS OF TABLE gt_sbooks
         FROM sbook
         WHERE carrid IN so_carid
         AND connid IN so_conid
         AND fldate IN so_fdate

         AND cancelled EQ p_can

         AND CLASS EQ 'Y'."이코노미 좌석 조회
  ENDCASE.
  IF SY-subrc <> 0.
    MESSAGE S001.
    EXIT.
   ENDIF.

  "SCARR 테이블을 참조하여 항공사 이름 필드 조회
  LOOP AT gt_sbooks INTO gs_sbook.
    SELECT SINGLE CARRNAME INTO GS_SBOOK-CARRNAME
      FROM SCARR
      WHERE CARRID = GS_SBOOK-CARRID.

    MODIFY gt_sbooks FROM gs_sbook.
   ENDLOOP.
ENDFORM.
*&---------------------------------------------------------------------*
*& Form MAKE_DATA
*&---------------------------------------------------------------------*
*& 데이터 가공 및 처리
*&---------------------------------------------------------------------*
*& -->  p1        text
*& <--  p2        text
*&---------------------------------------------------------------------*
FORM make_data .
  LOOP AT gt_sbooks INTO gs_sbook.
    "B- CUSTYPE 필드 값에 따라 LIGHT 필드 LED 지정
    IF gs_sbook-custtype = 'B'.
      gs_sbook-light = '3'. "GREEN
    ELSE.
      gs_sbook-light = '2'. "YELLOW
    ENDIF.

    "C- ROW COLOR 지정 - 흡연자 여부
    IF gs_sbook-smoker = 'X'. "흡연자일 경우, col_negative 색 적용
      gs_sbook-color = 'C' && col_negative && '10'.
    ENDIF.

    "D- CELL COLOR 지정 - INVOICE 선택 여부
    IF gs_sbook-invoice = 'X'.
      gs_field_col-fname = 'INVOICE'.
      gs_field_col-color-col = col_positive.
      gs_field_col-color-inv = 0.
      APPEND gs_field_col TO gs_sbook-field_col.
    ENDIF.

    "D- CANCELLED 여부에 따라 CANCEL_ICON 필드의 ICON 설정
    IF gs_sbook-cancelled = 'X'.
      gs_sbook-cancel_icon = icon_red_light.
    ELSE.
      gs_sbook-cancel_icon = icon_green_light.
    ENDIF.

    MODIFY gt_sbooks FROM gs_sbook
              TRANSPORTING light color field_col cancel_icon.
  ENDLOOP.
ENDFORM.
