REPORT yb03_select_test.

*** 선언부
DATA: BEGIN OF gs_sflight,
        carrid    TYPE sflight-carrid,
        connid    TYPE sflight-connid,
        fldate    TYPE sflight-fldate,
        planetype TYPE sflight-planetype,
        seatsmax  TYPE sflight-seatsmax,
      END OF gs_sflight,
      gt_sflight LIKE TABLE OF gs_sflight.

* select-options 조건
SELECT-OPTIONS so_carid FOR gs_sflight-carrid. "DEFAULT 'AA' to 'LH'.

" header line 포함 , 이름이 같아.
so_carid-sign = 'I'. " or E'.
so_carid-option = 'BT'.
so_carid-low = 'AA'.
so_carid-high = 'LH'.
APPEND so_carid.

* 입력 조건
PARAMETERS: gv_conid TYPE sflight-connid.

*** 구현부
* DB Tab.에서 필드 조회
SELECT carrid connid fldate planetype seatsmax
  INTO CORRESPONDING FIELDS OF TABLE gt_sflight
  FROM sflight
  where connid = gv_conid
  and carrid in so_carid.

*** 출력부
cl_demo_output=>display( gt_sflight ).
